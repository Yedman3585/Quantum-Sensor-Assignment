
import numpy as np
import pandas as pd
import time
import logging
from scipy.spatial import distance_matrix
import warnings
import os
import json
from datetime import datetime

warnings.filterwarnings('ignore')

try:
    import dimod
    from neal import SimulatedAnnealingSampler
    HAVE_DIMOD = True
except ImportError:
    HAVE_DIMOD = False

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('quantum_optimized.log', encoding='utf-8'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)


class OptimizedQuantumScheduler:
    assignment_matrix: None

    def __init__(self, n_cameras=20000, n_servers=800,
                 batch_size=80, max_servers_per_batch=20,
                 random_seed=42):

        logger.info(f"Initialization of cameras: {n_cameras} and servers: {n_servers} ")

        self.n_cameras = n_cameras
        self.n_servers = n_servers
        self.batch_size = batch_size
        self.max_servers_per_batch = max_servers_per_batch

        self.random_seed = random_seed
        np.random.seed(self.random_seed)

        self.cameras_df = None
        self.servers_df = None
        self.cost_matrix = None
        self.assignment_matrix = None
        self.remaining_capacity = None
        self.assigned_cameras = set()

        #----------------------------------------------------------
        self.log_dir = "logs"
        self.snapshot_dir = os.path.join(self.log_dir, "snapshots")
        os.makedirs(self.log_dir, exist_ok=True)
        os.makedirs(self.snapshot_dir, exist_ok=True)
        self.run_id = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.progress_log = os.path.join(self.log_dir, f"progress_{self.run_id}.jsonl")
        self.snapshot_every = 50
        #-----------------------------------------------------------

    def log_progress(self, batch_idx, batch_assigned, coverage, success_rate,
                      energy=None, best_energy=None, qubo_time=0, quantum_time=0,
                      assignments=None):
        """Логирует прогресс в JSONL"""
        log_entry = {
            "run_id": self.run_id,
            "timestamp": datetime.now().isoformat(),
            "batch_idx": int(batch_idx),
            "batch_assigned": int(batch_assigned),
            "coverage_percent": float(coverage),
            "qubo_success_rate": float(success_rate),
            "qubo_time_sec": float(qubo_time),
            "quantum_time_sec": float(quantum_time),
            "energy": float(energy) if energy is not None else None,
            "best_energy": float(best_energy) if best_energy is not None else None,
            "assignments": assignments or []
        }
        with open(self.progress_log, "a", encoding="utf-8") as f:
            f.write(json.dumps(log_entry) + "\n")

    def generate_camera_server_data(self):
        logger.info("===Data Generation ===")

        priorities = np.random.choice([3, 2, 1], size=self.n_cameras, p=[0.15, 0.25, 0.6])
        weights = np.zeros(self.n_cameras)
        loads = np.zeros(self.n_cameras)

        mask_p1 = priorities == 3
        mask_p2 = priorities == 2
        mask_p3 = priorities == 1

        weights[mask_p1] = np.random.uniform(4, 7, np.sum(mask_p1))
        loads[mask_p1] = np.random.uniform(8, 15, np.sum(mask_p1))

        weights[mask_p2] = np.random.uniform(2, 4, np.sum(mask_p2))
        loads[mask_p2] = np.random.uniform(4, 8, np.sum(mask_p2))

        weights[mask_p3] = np.random.uniform(0.4, 0.8, np.sum(mask_p3))
        loads[mask_p3] = np.random.uniform(1, 3, np.sum(mask_p3))

        camera_x = np.random.uniform(0, 1000, self.n_cameras)
        camera_y = np.random.uniform(0, 1000, self.n_cameras)

        self.cameras_df = pd.DataFrame({
            'camera_id': range(self.n_cameras),
            'priority': priorities,
            'weight_Mbps': weights,
            'load_GFLOPS': loads,
            'x': camera_x,
            'y': camera_y
        })

        server_x = np.random.uniform(0, 1000, self.n_servers)
        server_y = np.random.uniform(0, 1000, self.n_servers)

        server_types = np.random.choice([3, 2, 1], size=self.n_servers, p=[0.1, 0.3, 0.6])
        capacities = np.zeros(self.n_servers)

        capacities[server_types == 3] = np.random.uniform(800, 1000, np.sum(server_types == 3))
        capacities[server_types == 2] = np.random.uniform(400, 800, np.sum(server_types == 2))
        capacities[server_types == 1] = np.random.uniform(200, 400, np.sum(server_types == 1))

        self.servers_df = pd.DataFrame({
            'server_id': range(self.n_servers),
            'capacity_GFLOPS': capacities,
            'x': server_x,
            'y': server_y
        })

        self.build_cost_matrix()
        self.remaining_capacity = capacities.copy()

        total_load = loads.sum()
        total_capacity = capacities.sum()
        utilization = total_load / total_capacity * 100
        logger.info(
            f"Overall load: {total_load:.1f}, capacity: {total_capacity:.1f}, usage: {utilization:.1f}%")

        return utilization

    def build_cost_matrix(self):
        cam_coords = self.cameras_df[['x', 'y']].values
        srv_coords = self.servers_df[['x', 'y']].values

        distances = distance_matrix(cam_coords, srv_coords)
        distances_norm = distances / np.max(distances)

        priorities_norm = (3 - self.cameras_df['priority'].values) / 2.0
        loads_norm = self.cameras_df['load_GFLOPS'].values / np.max(self.cameras_df['load_GFLOPS'])

        capacities_inv = 1 / (self.servers_df['capacity_GFLOPS'].values + 1e-9)
        capacities_norm = capacities_inv / np.max(capacities_inv)

        self.cost_matrix = (
                0.40 * distances_norm +
                0.35 * loads_norm[:, None] +
                0.20 * priorities_norm[:, None] +
                0.05 * capacities_norm[None, :]
        )

        self.cost_matrix = (self.cost_matrix - np.min(self.cost_matrix)) / (
                np.max(self.cost_matrix) - np.min(self.cost_matrix) + 1e-9)

    def solve_with_quantum_annealing(self, num_reads=200):
        if not HAVE_DIMOD:
            logger.error("Dimod unavailable")
            return None, 0, 0, 0, 0

        logger.info("=== quantum annealing  ===")
        total_start = time.time()

        self.assignment_matrix = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        self.remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()
        self.assigned_cameras = set()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)
        total_batches = (self.n_cameras + self.batch_size - 1) // self.batch_size

        total_qubo_time = 0
        total_quantum_time = 0
        quantum_success_count = 0
        total_assigned = 0

        logger.info(f"processing {total_batches} batches...")

        self.current_batch_idx = 0

        for batch_idx in range(total_batches):
            self.current_batch_idx = batch_idx

            if len(self.assigned_cameras) / self.n_cameras > 0.995:
                logger.info(" 99.5% coverage achieved, completing")
                break

            start_idx = batch_idx * self.batch_size
            end_idx = min((batch_idx + 1) * self.batch_size, self.n_cameras)
            batch_indices = sorted_indices[start_idx:end_idx]

            available_indices = [idx for idx in batch_indices if idx not in self.assigned_cameras]
            if not available_indices:
                continue

            top_servers = self.select_servers_for_batch(available_indices)
            if len(top_servers) == 0:
                continue

            batch_complexity = self.calculate_batch_complexity(available_indices, top_servers)
            adaptive_reads = max(num_reads, int(num_reads * (1 + batch_complexity)))

            batch_solution, qubo_time, quantum_time, quantum_used = self.solve_batch_assignment(
                available_indices, top_servers, adaptive_reads
            )

            total_qubo_time += qubo_time
            total_quantum_time += quantum_time
            if quantum_used:
                quantum_success_count += 1

            batch_solution = self.improve_batch_solution(batch_solution, available_indices, top_servers)

            assignments_in_batch = []
            batch_assigned = 0

            for i, cam_idx in enumerate(available_indices):
                assigned = False
                for j_local, j_global in enumerate(top_servers):
                    if batch_solution[i, j_local] == 1 and not assigned:
                        cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                        if self.remaining_capacity[j_global] >= cam_load:
                            self.assignment_matrix[cam_idx, j_global] = 1
                            self.remaining_capacity[j_global] -= cam_load
                            self.assigned_cameras.add(cam_idx)
                            batch_assigned += 1
                            assigned = True
                            assignments_in_batch.append({
                                "cam_id": int(cam_idx),
                                "server_id": int(j_global)
                            })

            total_assigned += batch_assigned

            current_coverage = len(self.assigned_cameras) / self.n_cameras * 100
            success_rate = (quantum_success_count / (batch_idx + 1)) * 100 if batch_idx > 0 else 0

            self.log_progress(
                batch_idx=batch_idx,
                batch_assigned=batch_assigned,
                coverage=current_coverage,
                success_rate=success_rate,
                assignments=assignments_in_batch
            )

            if (batch_idx + 1) % self.snapshot_every == 0:
                snapshot_path = os.path.join(self.snapshot_dir, f"snapshot_{self.run_id}_batch_{batch_idx + 1}.npz")
                np.savez_compressed(
                    snapshot_path,
                    assignment=self.assignment_matrix,
                    cameras=self.cameras_df.values,
                    servers=self.servers_df.values,
                    remaining_capacity=self.remaining_capacity
                )
                logger.info(f"SNAPSHOT: {snapshot_path}")

            if (batch_idx + 1) % 20 == 0:
                logger.info(
                    f"Batch {batch_idx + 1}: {batch_assigned} cameras, "
                    f"coverage: {current_coverage:.1f}%, QUBO success: {success_rate:.1f}%"
                )

        total_time = time.time() - total_start
        self.optimize_assignments()
        objective = self.calculate_objective_function(self.assignment_matrix)

        self.analyze_performance_metrics(
            total_qubo_time, total_quantum_time, total_time,
            total_assigned, quantum_success_count, total_batches
        )

        return self.assignment_matrix, objective, total_time, total_qubo_time, total_quantum_time

    def calculate_batch_complexity(self, batch_indices, top_servers):
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        total_load = batch_loads.sum()
        available_capacity = self.remaining_capacity[top_servers].sum()

        complexity = min(1.0, total_load / (available_capacity + 1e-9))
        return complexity

    def select_servers_for_batch(self, batch_indices):
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        min_load = np.min(batch_loads)
        avg_load = np.mean(batch_loads)

        valid_servers = np.where(self.remaining_capacity >= avg_load * 0.7)[0]

        if len(valid_servers) == 0:
            valid_servers = np.where(self.remaining_capacity >= min_load)[0]

        if len(valid_servers) == 0:
            return np.array([])

        scores = []
        for server_idx in valid_servers:
            server_capacity = self.remaining_capacity[server_idx]

            avg_cost = np.mean(self.cost_matrix[batch_indices, server_idx])

            feasible_cameras = np.sum(batch_loads <= server_capacity)

            capacity_score = server_capacity / np.max(self.remaining_capacity[valid_servers])
            cost_score = 1 - avg_cost
            feasibility_score = feasible_cameras / len(batch_indices)

            total_score = capacity_score * 0.3 + cost_score * 0.5 + feasibility_score * 0.2
            scores.append(total_score)

        scores = np.array(scores)
        top_indices = np.argsort(-scores)[:self.max_servers_per_batch]

        return valid_servers[top_indices]

    def solve_batch_assignment(self, batch_indices, top_servers, num_reads):
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        qubo_start = time.time()
        Q = self.build_qubo_matrix(batch_indices, top_servers)
        qubo_time = time.time() - qubo_start

        if not Q:
            solution = self.solve_batch_greedy(batch_indices, top_servers)
            self.log_progress(
                batch_idx=self.current_batch_idx,
                batch_assigned=0,
                coverage=len(self.assigned_cameras) / self.n_cameras * 100,
                success_rate=0,
                qubo_time=qubo_time,
                quantum_time=0
            )
            return solution, qubo_time, 0, False

        try:
            quantum_start = time.time()
            sampler = SimulatedAnnealingSampler()
            response = sampler.sample_qubo(Q, num_reads=num_reads)
            quantum_time = time.time() - quantum_start

            best_energy = response.first.energy

            assignment = self.decode_quantum_solution(response, n_batch, top_servers, batch_indices)

            assignments = []
            for i, cam_idx in enumerate(batch_indices):
                for j, srv_idx in enumerate(top_servers):
                    if assignment[i, j] == 1:
                        assignments.append({"cam_id": int(cam_idx), "server_id": int(srv_idx)})

            valid_assignments = len(assignments)
            if valid_assignments >= n_batch * 0.5:
                self.log_progress(
                    batch_idx=self.current_batch_idx,
                    batch_assigned=valid_assignments,
                    coverage=len(self.assigned_cameras) / self.n_cameras * 100,
                    success_rate=100.0,
                    energy=best_energy,
                    best_energy=best_energy,
                    qubo_time=qubo_time,
                    quantum_time=quantum_time,
                    assignments=assignments
                )
                return assignment, qubo_time, quantum_time, True
            else:
                solution = self.solve_batch_greedy(batch_indices, top_servers)
                self.log_progress(
                    batch_idx=self.current_batch_idx,
                    batch_assigned=0,
                    coverage=len(self.assigned_cameras) / self.n_cameras * 100,
                    success_rate=0,
                    qubo_time=qubo_time,
                    quantum_time=quantum_time
                )
                return solution, qubo_time, quantum_time, False

        except Exception as e:
            logger.warning(f"Ошибка отжига: {e}")
            solution = self.solve_batch_greedy(batch_indices, top_servers)
            self.log_progress(
                batch_idx=self.current_batch_idx,
                batch_assigned=0,
                coverage=len(self.assigned_cameras) / self.n_cameras * 100,
                success_rate=0,
                qubo_time=qubo_time,
                quantum_time=0
            )
            return solution, qubo_time, 0, False

    def build_qubo_matrix(self, batch_indices, top_servers):
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        batch_costs = self.cost_matrix[batch_indices][:, top_servers]
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        batch_priorities = self.cameras_df.iloc[batch_indices]['priority'].values

        Q = {}

        for i in range(n_batch):
            priority_weight = (4 - batch_priorities[i])
            for j in range(n_servers_batch):
                var_name = f"x_{i}_{top_servers[j]}"
                cost = batch_costs[i, j]
                load = batch_loads[i]
                server_capacity = self.remaining_capacity[top_servers[j]]

                if load <= server_capacity:
                    reward = -25.0 * (1.0 - cost) * priority_weight
                    Q[(var_name, var_name)] = reward
                else:
                    penalty = 100.0
                    Q[(var_name, var_name)] = penalty

        for i in range(n_batch):
            for j1 in range(n_servers_batch):
                var1 = f"x_{i}_{top_servers[j1]}"
                for j2 in range(j1 + 1, n_servers_batch):
                    var2 = f"x_{i}_{top_servers[j2]}"
                    Q[(var1, var2)] = 15.0

        return Q

    def decode_quantum_solution(self, response, n_batch, top_servers, batch_indices):
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        best_solution = response.first.sample

        candidates = []
        for i in range(n_batch):
            for j, server_idx in enumerate(top_servers):
                var_name = f"x_{i}_{server_idx}"
                if best_solution.get(var_name, 0) == 1:
                    cost = self.cost_matrix[batch_indices[i], server_idx]
                    priority = self.cameras_df.iloc[batch_indices[i]]['priority']
                    load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']
                    server_capacity = self.remaining_capacity[server_idx]

                    capacity_utilization = 1.0 - (load / server_capacity) if server_capacity > 0 else 0
                    score = priority * 0.4 + (1 - cost) * 0.4 + capacity_utilization * 0.2
                    candidates.append((i, j, score, cost, priority))

        candidates.sort(key=lambda x: -x[2])

        used_cameras = set()
        server_loads = {server_idx: 0 for server_idx in top_servers}

        for i, j, score, cost, priority in candidates:
            if i not in used_cameras:
                server_idx = top_servers[j]
                cam_idx = batch_indices[i]
                cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

                if server_loads[server_idx] + cam_load <= self.remaining_capacity[server_idx]:
                    assignment[i, j] = 1
                    used_cameras.add(i)
                    server_loads[server_idx] += cam_load

        return assignment

    def improve_batch_solution(self, assignment, batch_indices, top_servers):
        improved = assignment.copy()
        n_batch = len(batch_indices)

        for i in range(n_batch):
            current_server = None
            current_cost = float('inf')

            for j in range(len(top_servers)):
                if assignment[i, j] == 1:
                    current_server = j
                    current_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                    break

            if current_server is not None:
                for j in range(len(top_servers)):
                    if j != current_server:
                        new_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                        cam_load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']

                        if (new_cost < current_cost * 0.95 and
                                self.remaining_capacity[top_servers[j]] >= cam_load):
                            improved[i, current_server] = 0
                            improved[i, j] = 1
                            break

        return improved

    def optimize_assignments(self):
        logger.info("Final optimization...")

        for iteration in range(3):
            improvements = 0
            for i in range(self.n_cameras):
                current_server = np.argmax(self.assignment_matrix[i]) if np.any(self.assignment_matrix[i]) else -1
                if current_server == -1:
                    continue

                current_cost = self.cost_matrix[i, current_server]
                current_load = self.cameras_df.iloc[i]['load_GFLOPS']

                best_server = current_server
                best_cost = current_cost

                for j in range(self.n_servers):
                    if j != current_server:
                        new_cost = self.cost_matrix[i, j]
                        if (new_cost < best_cost * 0.98 and
                                self.remaining_capacity[j] >= current_load):

                            if self.can_reassign_camera(i, current_server, j):
                                best_server = j
                                best_cost = new_cost

                if best_server != current_server:
                    self.assignment_matrix[i, current_server] = 0
                    self.assignment_matrix[i, best_server] = 1
                    self.remaining_capacity[current_server] += current_load
                    self.remaining_capacity[best_server] -= current_load
                    improvements += 1

            if improvements == 0:
                break

            logger.info(f"Iteration {iteration + 1}: {improvements} improvements")

    def can_reassign_camera(self, cam_idx, from_server, to_server):
        cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
        return self.remaining_capacity[to_server] >= cam_load

    def solve_batch_greedy(self, batch_indices, top_servers):
        n_batch = len(batch_indices)
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)

        remaining_capacity = self.remaining_capacity[top_servers].copy()

        batch_data = self.cameras_df.iloc[batch_indices]
        priority_scores = batch_data['priority'].values * batch_data['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for pos in sorted_indices:
            cam_idx = batch_indices[pos]
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j, server_idx in enumerate(top_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, server_idx]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[server_idx]['capacity_GFLOPS']

                    score = (cam_priority * 0.5) + ((1 - cost) * 0.3) + (capacity_ratio * 0.2)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[pos, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        return assignment

    def solve_with_greedy_timeout(self, timeout_seconds=180):
        """Жадный алгоритм с УМНЫМ таймаутом - останавливается только при зависании"""
        logger.info("=== optimized greedy algorithm (with smart timeout) ===")
        start_time = time.time()
        last_progress_time = start_time

        assignment = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        processed = 0
        for idx, cam_idx in enumerate(sorted_indices):
            current_time = time.time()

            if current_time - last_progress_time > 30:
                logger.warning(
                    f"Greedy algorithm STALLED - no progress for 30 seconds. Processed {idx}/{self.n_cameras} cameras")
                break

            if current_time - start_time > timeout_seconds:
                logger.warning(
                    f"Greedy algorithm TIMEOUT after {timeout_seconds} sec. Processed {idx}/{self.n_cameras} cameras")
                break

            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j in range(self.n_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, j]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[j]['capacity_GFLOPS']

                    score = (cam_priority * 0.6) + ((1 - cost) * 0.4) + (capacity_ratio * 0.1)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[cam_idx, best_server] = 1
                remaining_capacity[best_server] -= cam_load
                processed += 1
                last_progress_time = current_time

            if idx % 1000 == 0:
                logger.info(f"Greedy progress: {idx}/{self.n_cameras} cameras processed")

        time_taken = time.time() - start_time
        objective = self.calculate_objective_function(assignment)
        covered = np.sum(np.any(assignment, axis=1))

        logger.info(
            f"greedy algorithm: {time_taken:.3f}sec, {covered} cameras (processed {processed}/{self.n_cameras})")
        return assignment, objective, time_taken

    def calculate_objective_function(self, assignment):
        """ОПТИМИЗИРОВАННАЯ целевая функция"""
        total_cost = 0
        uncovered_penalty = 0
        overload_penalty = 0

        assigned_mask = np.any(assignment, axis=1)
        assigned_indices = np.where(assigned_mask)[0]

        for i in assigned_indices:
            j = np.argmax(assignment[i])
            priority = self.cameras_df.iloc[i]['priority']
            priority_weight = (4 - priority)
            total_cost += self.cost_matrix[i, j] * priority_weight

        uncovered = self.n_cameras - np.sum(assigned_mask)
        uncovered_penalty = uncovered * 15.0

        for j in range(self.n_servers):
            server_load = np.sum(self.cameras_df['load_GFLOPS'].values * assignment[:, j])
            overload = max(0, server_load - self.servers_df.iloc[j]['capacity_GFLOPS'])
            overload_penalty += overload * 8.0

        total_objective = total_cost + uncovered_penalty + overload_penalty

        logger.info(
            f"Objective: cost={total_cost:.1f}, uncovered={uncovered_penalty:.1f}, overload={overload_penalty:.1f}")

        return total_objective

class OptimizedQuantumScheduler:
    def __init__(self, n_cameras=5000, n_servers=250,
                 batch_size=30, max_servers_per_batch=20,
                 random_seed=42):

        logger.info(f"Initialization of cameras: {n_cameras} and servers: {n_servers} ")

        self.n_cameras = n_cameras

    def generate_realistic_data(self):

        logger.info("=== ГЕНЕРАЦИЯ ДАННЫХ ===")

        priorities = np.random.choice([3, 2, 1], size=self.n_cameras, p=[0.15, 0.25, 0.6])
        weights = np.zeros(self.n_cameras)
        loads = np.zeros(self.n_cameras)

        mask_p1 = priorities == 3
        mask_p2 = priorities == 2
        mask_p3 = priorities == 1

        weights[mask_p1] = np.random.uniform(4, 7, np.sum(mask_p1))
        loads[mask_p1] = np.random.uniform(8, 15, np.sum(mask_p1))

        weights[mask_p2] = np.random.uniform(2, 4, np.sum(mask_p2))
        loads[mask_p2] = np.random.uniform(4, 8, np.sum(mask_p2))

        weights[mask_p3] = np.random.uniform(0.4, 0.8, np.sum(mask_p3))
        loads[mask_p3] = np.random.uniform(1, 3, np.sum(mask_p3))

        camera_x = np.random.uniform(0, 1000, self.n_cameras)
        camera_y = np.random.uniform(0, 1000, self.n_cameras)

        self.cameras_df = pd.DataFrame({
            'camera_id': range(self.n_cameras),
            'priority': priorities,
            'weight_Mbps': weights,
            'load_GFLOPS': loads,
            'x': camera_x,
            'y': camera_y
        })

        server_x = np.random.uniform(0, 1000, self.n_servers)
        server_y = np.random.uniform(0, 1000, self.n_servers)

        server_types = np.random.choice([3, 2, 1], size=self.n_servers, p=[0.1, 0.3, 0.6])
        capacities = np.zeros(self.n_servers)

        capacities[server_types == 3] = np.random.uniform(800, 1000, np.sum(server_types == 3))
        capacities[server_types == 2] = np.random.uniform(400, 800, np.sum(server_types == 2))
        capacities[server_types == 1] = np.random.uniform(200, 400, np.sum(server_types == 1))

        self.servers_df = pd.DataFrame({
            'server_id': range(self.n_servers),
            'capacity_GFLOPS': capacities,
            'x': server_x,
            'y': server_y
        })

        self._build_cost_matrix()
        self.remaining_capacity = capacities.copy()

        total_load = loads.sum()
        total_capacity = capacities.sum()
        utilization = total_load / total_capacity * 100
        logger.info(
            f"Общая нагрузка: {total_load:.1f}, емкость: {total_capacity:.1f}, использование: {utilization:.1f}%")

        return utilization

    def _build_cost_matrix(self):
        """Матрица стоимости"""
        cam_coords = self.cameras_df[['x', 'y']].values
        srv_coords = self.servers_df[['x', 'y']].values

        distances = distance_matrix(cam_coords, srv_coords)
        distances_norm = distances / np.max(distances)

        priorities_norm = (3 - self.cameras_df['priority'].values) / 2.0
        loads_norm = self.cameras_df['load_GFLOPS'].values / np.max(self.cameras_df['load_GFLOPS'])

        capacities_inv = 1 / (self.servers_df['capacity_GFLOPS'].values + 1e-9)
        capacities_norm = capacities_inv / np.max(capacities_inv)



        # self.cost_matrix = (
        #         0.6 * distances_norm +  # Больший вес расстояниям
        #         0.2 * loads_norm[:, None] +  # Меньший вес нагрузке
        #         0.1 * priorities_norm[:, None] +  # Меньший вес приоритету
        #         0.1 * capacities_norm[None, :]  # Меньший вес емкости
        # )

        self.cost_matrix = (
                0.55 * distances_norm +  # Оптимальный вес расстояниям
                0.25 * loads_norm[:, None] +  # Увеличенный вес нагрузке
                0.15 * priorities_norm[:, None] +  # Больше внимания приоритету
                0.05 * capacities_norm[None, :]  # Минимальный вес емкости
        )

        self.cost_matrix = (self.cost_matrix - np.min(self.cost_matrix)) / (
                    np.max(self.cost_matrix) - np.min(self.cost_matrix) + 1e-9)



    # def solve_with_quantum_optimized(self, num_reads=150):
    def solve_with_quantum_optimized(self, num_reads=200):
        # Больше чтений для качества


        """ОПТИМИЗИРОВАННЫЙ квантовый отжиг"""
        if not HAVE_DIMOD:
            logger.error("Dimod недоступен")
            return None, 0, 0, 0, 0

        logger.info("=== ОПТИМИЗИРОВАННЫЙ КВАНТОВЫЙ ОТЖИГ ===")
        total_start = time.time()

        self.assignment_matrix = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        self.remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()
        self.assigned_cameras = set()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)
        total_batches = (self.n_cameras + self.batch_size - 1) // self.batch_size

        total_qubo_time = 0
        total_quantum_time = 0
        quantum_success_count = 0
        total_assigned = 0

        logger.info(f"Обработка {total_batches} батчей...")

        for batch_idx in range(total_batches):
            if len(self.assigned_cameras) / self.n_cameras > 0.995:
                logger.info(f"Достигнуто 99.5% покрытие, завершаем")
                break

            start_idx = batch_idx * self.batch_size
            end_idx = min((batch_idx + 1) * self.batch_size, self.n_cameras)
            batch_indices = sorted_indices[start_idx:end_idx]

            available_indices = [idx for idx in batch_indices if idx not in self.assigned_cameras]
            if not available_indices:
                continue

            top_servers = self._select_servers_optimized(available_indices)
            if len(top_servers) == 0:
                continue

            # Адаптивные чтения: больше для сложных батчей
            batch_complexity = self._calculate_batch_complexity(available_indices, top_servers)
            adaptive_reads = max(num_reads, int(num_reads * (1 + batch_complexity)))

            batch_solution, qubo_time, quantum_time, quantum_used = self._solve_batch_optimized(
                available_indices, top_servers, adaptive_reads
            )

            total_qubo_time += qubo_time
            total_quantum_time += quantum_time
            if quantum_used:
                quantum_success_count += 1

            # Пост-обработка решения
            batch_solution = self._post_process_solution(batch_solution, available_indices, top_servers)

            batch_assigned = 0
            for i, cam_idx in enumerate(available_indices):
                assigned = False
                for j_local, j_global in enumerate(top_servers):
                    if batch_solution[i, j_local] == 1 and not assigned:
                        if self.remaining_capacity[j_global] >= self.cameras_df.iloc[cam_idx]['load_GFLOPS']:
                            self.assignment_matrix[cam_idx, j_global] = 1
                            self.remaining_capacity[j_global] -= self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                            self.assigned_cameras.add(cam_idx)
                            batch_assigned += 1
                            assigned = True

            total_assigned += batch_assigned

            if (batch_idx + 1) % 20 == 0:
                current_coverage = len(self.assigned_cameras) / self.n_cameras * 100
                success_rate = quantum_success_count / (batch_idx + 1) * 100
                logger.info(
                    f"Батч {batch_idx + 1}: {batch_assigned} камер, покрытие: {current_coverage:.1f}%, успех QUBO: {success_rate:.1f}%")

        total_time = time.time() - total_start

        # Финальная оптимизация решения
        self._optimize_final_solution()

        objective = self._calculate_optimized_objective(self.assignment_matrix)

        self._analyze_optimized_performance(total_qubo_time, total_quantum_time, total_time,
                                            total_assigned, quantum_success_count, total_batches)

        return self.assignment_matrix, objective, total_time, total_qubo_time, total_quantum_time

    def _calculate_batch_complexity(self, batch_indices, top_servers):
        """Оценка сложности батча"""
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        total_load = batch_loads.sum()
        available_capacity = self.remaining_capacity[top_servers].sum()

        # Сложность = насколько близка нагрузка к доступной емкости
        complexity = min(1.0, total_load / (available_capacity + 1e-9))
        return complexity

    def _select_servers_optimized(self, batch_indices):
        """Оптимизированный выбор серверов"""
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        min_load = np.min(batch_loads)
        avg_load = np.mean(batch_loads)

        # Выбираем серверы с достаточной емкостью для средней нагрузки
        valid_servers = np.where(self.remaining_capacity >= avg_load * 0.7)[0]

        if len(valid_servers) == 0:
            # Если нет подходящих, берем с минимальной нагрузкой
            valid_servers = np.where(self.remaining_capacity >= min_load)[0]

        if len(valid_servers) == 0:
            return np.array([])

        scores = []
        for server_idx in valid_servers:
            server_capacity = self.remaining_capacity[server_idx]

            # Средняя стоимость для камер батча
            avg_cost = np.mean(self.cost_matrix[batch_indices, server_idx])

            # Количество камер, которые могут быть размещены
            feasible_cameras = np.sum(batch_loads <= server_capacity)

            capacity_score = server_capacity / np.max(self.remaining_capacity[valid_servers])
            cost_score = 1 - avg_cost
            feasibility_score = feasible_cameras / len(batch_indices)

            # Оптимизированная формула
            total_score = capacity_score * 0.3 + cost_score * 0.5 + feasibility_score * 0.2
            scores.append(total_score)

        scores = np.array(scores)
        top_indices = np.argsort(-scores)[:self.max_servers_per_batch]

        return valid_servers[top_indices]

    def _solve_batch_optimized(self, batch_indices, top_servers, num_reads):
        """Оптимизированное решение батча"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        qubo_start = time.time()
        Q = self._build_optimized_qubo(batch_indices, top_servers)
        qubo_time = time.time() - qubo_start

        if not Q:
            solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
            return solution, qubo_time, 0, False

        try:
            quantum_start = time.time()
            sampler = SimulatedAnnealingSampler()
            response = sampler.sample_qubo(Q, num_reads=num_reads)
            quantum_time = time.time() - quantum_start

            assignment = self._decode_optimized_solution(response, n_batch, top_servers, batch_indices)

            # Более строгая проверка качества
            valid_assignments = np.sum(assignment)
            if valid_assignments >= n_batch * 0.5:  # 50% успешных назначений
                return assignment, qubo_time, quantum_time, True
            else:
                solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
                return solution, qubo_time, quantum_time, False

        except Exception as e:
            logger.warning(f"Ошибка отжига: {e}, используем жадный алгоритм")
            solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
            return solution, qubo_time, 0, False

    def _build_optimized_qubo(self, batch_indices, top_servers):
        """ОПТИМИЗИРОВАННАЯ QUBO матрица"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        batch_costs = self.cost_matrix[batch_indices][:, top_servers]
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        batch_priorities = self.cameras_df.iloc[batch_indices]['priority'].values

        Q = {}

        # УЛУЧШЕННЫЕ НАГРАДЫ: более агрессивные за хорошие назначения
        for i in range(n_batch):
            priority_weight = (4 - batch_priorities[i])  # 3->1, 2->2, 1->3
            for j in range(n_servers_batch):
                var_name = f"x_{i}_{top_servers[j]}"
                cost = batch_costs[i, j]
                load = batch_loads[i]
                server_capacity = self.remaining_capacity[top_servers[j]]

                if load <= server_capacity:
                    # БОЛЬШАЯ награда за хорошие назначения
                    reward = -25.0 * (1.0 - cost) * priority_weight
                    Q[(var_name, var_name)] = reward
                else:
                    # БОЛЬШОЙ штраф за невозможные назначения
                    penalty = 100.0
                    Q[(var_name, var_name)] = penalty

        # БОЛЕЕ СИЛЬНЫЕ ограничения против конфликтов
        for i in range(n_batch):
            for j1 in range(n_servers_batch):
                var1 = f"x_{i}_{top_servers[j1]}"
                for j2 in range(j1 + 1, n_servers_batch):
                    var2 = f"x_{i}_{top_servers[j2]}"
                    Q[(var1, var2)] = 15.0  # Увеличенный штраф

        return Q

    def _decode_optimized_solution(self, response, n_batch, top_servers, batch_indices):
        """Оптимизированное декодирование"""
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        best_solution = response.first.sample

        # УЛУЧШЕННЫЙ алгоритм: учитываем стоимость и емкость
        candidates = []
        for i in range(n_batch):
            for j, server_idx in enumerate(top_servers):
                var_name = f"x_{i}_{server_idx}"
                if best_solution.get(var_name, 0) == 1:
                    cost = self.cost_matrix[batch_indices[i], server_idx]
                    priority = self.cameras_df.iloc[batch_indices[i]]['priority']
                    load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']
                    server_capacity = self.remaining_capacity[server_idx]

                    # Score: выше приоритет, ниже стоимость, лучше использование емкости
                    capacity_utilization = 1.0 - (load / server_capacity) if server_capacity > 0 else 0
                    score = priority * 0.4 + (1 - cost) * 0.4 + capacity_utilization * 0.2
                    candidates.append((i, j, score, cost, priority))

        # Сортируем по score
        candidates.sort(key=lambda x: -x[2])

        used_cameras = set()
        server_loads = {server_idx: 0 for server_idx in top_servers}

        for i, j, score, cost, priority in candidates:
            if i not in used_cameras:
                server_idx = top_servers[j]
                cam_idx = batch_indices[i]
                cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

                if server_loads[server_idx] + cam_load <= self.remaining_capacity[server_idx]:
                    assignment[i, j] = 1
                    used_cameras.add(i)
                    server_loads[server_idx] += cam_load

        return assignment

    def _post_process_solution(self, assignment, batch_indices, top_servers):
        """Пост-обработка решения для улучшения качества"""
        improved = assignment.copy()
        n_batch = len(batch_indices)

        # Локальный поиск: пробуем улучшить каждое назначение
        for i in range(n_batch):
            current_server = None
            current_cost = float('inf')

            # Находим текущее назначение
            for j in range(len(top_servers)):
                if assignment[i, j] == 1:
                    current_server = j
                    current_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                    break

            if current_server is not None:
                # Ищем лучшее назначение
                for j in range(len(top_servers)):
                    if j != current_server:
                        new_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                        cam_load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']

                        # Проверяем улучшение и допустимость
                        if (new_cost < current_cost * 0.95 and
                                self.remaining_capacity[top_servers[j]] >= cam_load):
                            improved[i, current_server] = 0
                            improved[i, j] = 1
                            break

        return improved

    def _optimize_final_solution(self):
        """Финальная оптимизация всего решения"""
        logger.info("Применение финальной оптимизации...")

        # Перераспределяем камеры для уменьшения стоимости
        for iteration in range(3):  # 3 итерации оптимизации
            improvements = 0
            for i in range(self.n_cameras):
                current_server = np.argmax(self.assignment_matrix[i]) if np.any(self.assignment_matrix[i]) else -1
                if current_server == -1:
                    continue

                current_cost = self.cost_matrix[i, current_server]
                current_load = self.cameras_df.iloc[i]['load_GFLOPS']

                # Ищем лучший сервер
                best_server = current_server
                best_cost = current_cost

                for j in range(self.n_servers):
                    if j != current_server:
                        new_cost = self.cost_matrix[i, j]
                        if (new_cost < best_cost * 0.98 and  # Значительное улучшение
                                self.remaining_capacity[j] >= current_load):

                            # Проверяем, что перераспределение возможно
                            if self._can_reassign(i, current_server, j):
                                best_server = j
                                best_cost = new_cost

                if best_server != current_server:
                    # Выполняем перераспределение
                    self.assignment_matrix[i, current_server] = 0
                    self.assignment_matrix[i, best_server] = 1
                    self.remaining_capacity[current_server] += current_load
                    self.remaining_capacity[best_server] -= current_load
                    improvements += 1

            if improvements == 0:
                break

            logger.info(f"Итерация {iteration + 1}: {improvements} улучшений")

    def _can_reassign(self, cam_idx, from_server, to_server):
        """Проверка возможности перераспределения"""
        cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
        return self.remaining_capacity[to_server] >= cam_load

    def _solve_batch_greedy_optimized(self, batch_indices, top_servers):
        """Оптимизированный жадный алгоритм"""
        n_batch = len(batch_indices)
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)

        remaining_capacity = self.remaining_capacity[top_servers].copy()

        batch_data = self.cameras_df.iloc[batch_indices]
        priority_scores = batch_data['priority'].values * batch_data['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for pos in sorted_indices:
            cam_idx = batch_indices[pos]
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j, server_idx in enumerate(top_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, server_idx]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[server_idx]['capacity_GFLOPS']

                    # Score: выше приоритет, ниже стоимость, лучше баланс нагрузки
                    score = (cam_priority * 0.5) + ((1 - cost) * 0.3) + (capacity_ratio * 0.2)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[pos, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        return assignment

    def solve_with_greedy_optimized(self):
        """Оптимизированный жадный алгоритм"""
        logger.info("=== ОПТИМИЗИРОВАННЫЙ ЖАДНЫЙ АЛГОРИТМ ===")
        start_time = time.time()

        assignment = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for cam_idx in sorted_indices:
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j in range(self.n_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, j]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[j]['capacity_GFLOPS']

                    score = (cam_priority * 0.6) + ((1 - cost) * 0.4) + (capacity_ratio * 0.1)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[cam_idx, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        time_taken = time.time() - start_time
        objective = self._calculate_optimized_objective(assignment)
        covered = np.sum(np.any(assignment, axis=1))

        logger.info(f"Жадный алгоритм: {time_taken:.3f}с, {covered} камер")

        return assignment, objective, time_taken

    def _calculate_optimized_objective(self, assignment):
        """Оптимизированная целевая функция"""
        total_cost = 0
        uncovered_penalty = 0
        overload_penalty = 0


        for i in range(self.n_cameras):
            for j in range(self.n_servers):
                if assignment[i, j] == 1:
                    priority = self.cameras_df.iloc[i]['priority']
                    priority_weight = (4 - priority)
                    total_cost += self.cost_matrix[i, j] * priority_weight

        uncovered = self.n_cameras - np.sum(np.any(assignment, axis=1))
        uncovered_penalty = uncovered * 15.0

        for j in range(self.n_servers):
            server_load = np.sum(self.cameras_df['load_GFLOPS'].values * assignment[:, j])
            overload = max(0, server_load - self.servers_df.iloc[j]['capacity_GFLOPS'])
            overload_penalty += overload * 8.0

        total_objective = total_cost + uncovered_penalty + overload_penalty

        logger.info(
            f"Objective: cost={total_cost:.1f}, uncovered={uncovered_penalty:.1f}, overload={overload_penalty:.1f}")

        return total_objective

    def _analyze_optimized_performance(self, total_qubo_time, total_quantum_time, total_time,
                                       total_assigned, quantum_success, total_batches):
        """Анализ производительности"""
        logger.info("\n" + "=" * 50)
        logger.info("АНАЛИЗ ОПТИМИЗИРОВАННЫХ РЕЗУЛЬТАТОВ")
        logger.info("=" * 50)

        coverage = total_assigned / self.n_cameras * 100
        quantum_efficiency = (total_quantum_time / total_time * 100) if total_time > 0 else 0

        logger.info(
            f"Успешных QUBO батчей: {quantum_success}/{total_batches} ({quantum_success / total_batches * 100:.1f}%)")
        logger.info(f"Покрытие: {total_assigned}/{self.n_cameras} ({coverage:.1f}%)")
        logger.info(f"Общее время: {total_time:.2f}с")
        logger.info(f"Время QUBO: {total_qubo_time:.2f}с ({total_qubo_time / total_time * 100:.1f}%)")
        logger.info(f"Время отжига: {total_quantum_time:.2f}с ({quantum_efficiency:.1f}%)")

    def run_optimized_comparison(self):
        """Оптимизированное сравнение"""
        logger.info("=== ОПТИМИЗИРОВАННОЕ СРАВНЕНИЕ ===")

        results = {}

        logger.info("\n1. Запуск оптимизированного жадного алгоритма...")
        greedy_assignment, greedy_objective, greedy_time = self.solve_with_greedy_optimized()
        results['Greedy-Optimized'] = {
            'assignment': greedy_assignment,
            'objective': greedy_objective,
            'time': greedy_time,
            'covered': np.sum(np.any(greedy_assignment, axis=1))
        }

        if HAVE_DIMOD:
            logger.info("\n2. Запуск оптимизированного квантового отжига...")
            quantum_assignment, quantum_objective, quantum_time, quantum_qubo, quantum_annealing = \
                self.solve_with_quantum_optimized(num_reads=150)

            results['Quantum-Optimized'] = {
                'assignment': quantum_assignment,
                'objective': quantum_objective,
                'time': quantum_time,
                'qubo_time': quantum_qubo,
                'quantum_time': quantum_annealing,
                'covered': np.sum(np.any(quantum_assignment, axis=1))
            }

        self._print_optimized_results(results)
        return results

    def _print_optimized_results(self, results):
        """Печать оптимизированных результатов"""
        print("\n" + "=" * 100)
        print("ОПТИМИЗИРОВАННЫЕ РЕЗУЛЬТАТЫ")
        print("=" * 100)
        print(f"{'МЕТОД':<20} {'ВРЕМЯ (с)':<12} {'ЦЕЛЕВАЯ':<15} {'КАМЕРЫ':<12} {'ЭФФ. (%)':<12} {'УЛУЧШ.'}")
        print("-" * 100)

        base_objective = results['Greedy-Optimized']['objective']

        for method, data in results.items():
            if 'quantum_time' in data:
                efficiency = (data['quantum_time'] / data['time'] * 100) if data['time'] > 0 else 0
                improvement = ((base_objective - data['objective']) / base_objective * 100) if base_objective > 0 else 0
                improvement_str = f"{improvement:+.1f}%" if improvement != 0 else "0.0%"

                print(

            else:
                print(
                    f"{method:<20} {data['time']:<12.2f} {data['objective']:<15.1f} {data['covered']:<12} {'-':<12} {'-':<10}")

        print("=" * 100)

class OptimizedQuantumScheduler:
    def __init__(self, n_cameras=5000, n_servers=250,
                 batch_size=30, max_servers_per_batch=20,
                 random_seed=42):

        logger.info(f"Initialization of cameras: {n_cameras} and servers: {n_servers} ")

        self.n_cameras = n_cameras
        self.n_servers = n_servers
        self.batch_size = batch_size
        self.max_servers_per_batch = max_servers_per_batch

        self.random_seed = random_seed
        np.random.seed(self.random_seed)

        self.cameras_df = None
        self.servers_df = None

    def generate_realistic_data(self):

        logger.info("=== ГЕНЕРАЦИЯ ДАННЫХ ===")

        priorities = np.random.choice([3, 2, 1], size=self.n_cameras, p=[0.15, 0.25, 0.6])
        weights = np.zeros(self.n_cameras)
        loads = np.zeros(self.n_cameras)

        mask_p1 = priorities == 3
        mask_p2 = priorities == 2
        mask_p3 = priorities == 1

        weights[mask_p1] = np.random.uniform(4, 7, np.sum(mask_p1))
        loads[mask_p1] = np.random.uniform(8, 15, np.sum(mask_p1))

        weights[mask_p2] = np.random.uniform(2, 4, np.sum(mask_p2))
        loads[mask_p2] = np.random.uniform(4, 8, np.sum(mask_p2))

        weights[mask_p3] = np.random.uniform(0.4, 0.8, np.sum(mask_p3))
        loads[mask_p3] = np.random.uniform(1, 3, np.sum(mask_p3))

        camera_x = np.random.uniform(0, 1000, self.n_cameras)
        camera_y = np.random.uniform(0, 1000, self.n_cameras)

        self.cameras_df = pd.DataFrame({
            'camera_id': range(self.n_cameras),
            'priority': priorities,
            'weight_Mbps': weights,
            'load_GFLOPS': loads,
            'x': camera_x,
            'y': camera_y
        })

        server_x = np.random.uniform(0, 1000, self.n_servers)
        server_y = np.random.uniform(0, 1000, self.n_servers)

        server_types = np.random.choice([3, 2, 1], size=self.n_servers, p=[0.1, 0.3, 0.6])
        capacities = np.zeros(self.n_servers)

        capacities[server_types == 3] = np.random.uniform(800, 1000, np.sum(server_types == 3))
        capacities[server_types == 2] = np.random.uniform(400, 800, np.sum(server_types == 2))
        capacities[server_types == 1] = np.random.uniform(200, 400, np.sum(server_types == 1))

        self.servers_df = pd.DataFrame({
            'server_id': range(self.n_servers),
            'capacity_GFLOPS': capacities,
            'x': server_x,
            'y': server_y
        })

        self._build_cost_matrix()
        self.remaining_capacity = capacities.copy()

        total_load = loads.sum()
        total_capacity = capacities.sum()
        utilization = total_load / total_capacity * 100
        logger.info(
            f"Общая нагрузка: {total_load:.1f}, емкость: {total_capacity:.1f}, использование: {utilization:.1f}%")

        return utilization

    def _build_cost_matrix(self):
        """Матрица стоимости"""
        cam_coords = self.cameras_df[['x', 'y']].values
        srv_coords = self.servers_df[['x', 'y']].values

        distances = distance_matrix(cam_coords, srv_coords)
        distances_norm = distances / np.max(distances)

        priorities_norm = (3 - self.cameras_df['priority'].values) / 2.0
        loads_norm = self.cameras_df['load_GFLOPS'].values / np.max(self.cameras_df['load_GFLOPS'])

        capacities_inv = 1 / (self.servers_df['capacity_GFLOPS'].values + 1e-9)
        capacities_norm = capacities_inv / np.max(capacities_inv)



        # self.cost_matrix = (
        #         0.6 * distances_norm +  # Больший вес расстояниям
        #         0.2 * loads_norm[:, None] +  # Меньший вес нагрузке
        #         0.1 * priorities_norm[:, None] +  # Меньший вес приоритету
        #         0.1 * capacities_norm[None, :]  # Меньший вес емкости
        # )

        self.cost_matrix = (
                0.55 * distances_norm +  # Оптимальный вес расстояниям
                0.25 * loads_norm[:, None] +  # Увеличенный вес нагрузке
                0.15 * priorities_norm[:, None] +  # Больше внимания приоритету
                0.05 * capacities_norm[None, :]  # Минимальный вес емкости
        )

        self.cost_matrix = (self.cost_matrix - np.min(self.cost_matrix)) / (
                    np.max(self.cost_matrix) - np.min(self.cost_matrix) + 1e-9)



    # def solve_with_quantum_optimized(self, num_reads=150):
    def solve_with_quantum_optimized(self, num_reads=200):
        # Больше чтений для качества


        """ОПТИМИЗИРОВАННЫЙ квантовый отжиг"""
        if not HAVE_DIMOD:
            logger.error("Dimod недоступен")
            return None, 0, 0, 0, 0

        logger.info("=== ОПТИМИЗИРОВАННЫЙ КВАНТОВЫЙ ОТЖИГ ===")
        total_start = time.time()

        self.assignment_matrix = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        self.remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()
        self.assigned_cameras = set()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)
        total_batches = (self.n_cameras + self.batch_size - 1) // self.batch_size

        total_qubo_time = 0
        total_quantum_time = 0
        quantum_success_count = 0
        total_assigned = 0

        logger.info(f"Обработка {total_batches} батчей...")

        for batch_idx in range(total_batches):
            if len(self.assigned_cameras) / self.n_cameras > 0.995:
                logger.info(f"Достигнуто 99.5% покрытие, завершаем")
                break

            start_idx = batch_idx * self.batch_size
            end_idx = min((batch_idx + 1) * self.batch_size, self.n_cameras)
            batch_indices = sorted_indices[start_idx:end_idx]

            available_indices = [idx for idx in batch_indices if idx not in self.assigned_cameras]
            if not available_indices:
                continue

            top_servers = self._select_servers_optimized(available_indices)
            if len(top_servers) == 0:
                continue

            # Адаптивные чтения: больше для сложных батчей
            batch_complexity = self._calculate_batch_complexity(available_indices, top_servers)
            adaptive_reads = max(num_reads, int(num_reads * (1 + batch_complexity)))

            batch_solution, qubo_time, quantum_time, quantum_used = self._solve_batch_optimized(
                available_indices, top_servers, adaptive_reads
            )

            total_qubo_time += qubo_time
            total_quantum_time += quantum_time
            if quantum_used:
                quantum_success_count += 1

            # Пост-обработка решения
            batch_solution = self._post_process_solution(batch_solution, available_indices, top_servers)

            batch_assigned = 0
            for i, cam_idx in enumerate(available_indices):
                assigned = False
                for j_local, j_global in enumerate(top_servers):
                    if batch_solution[i, j_local] == 1 and not assigned:
                        if self.remaining_capacity[j_global] >= self.cameras_df.iloc[cam_idx]['load_GFLOPS']:
                            self.assignment_matrix[cam_idx, j_global] = 1
                            self.remaining_capacity[j_global] -= self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                            self.assigned_cameras.add(cam_idx)
                            batch_assigned += 1
                            assigned = True

            total_assigned += batch_assigned

            if (batch_idx + 1) % 20 == 0:
                current_coverage = len(self.assigned_cameras) / self.n_cameras * 100
                success_rate = quantum_success_count / (batch_idx + 1) * 100
                logger.info(
                    f"Батч {batch_idx + 1}: {batch_assigned} камер, покрытие: {current_coverage:.1f}%, успех QUBO: {success_rate:.1f}%")

        total_time = time.time() - total_start

        # Финальная оптимизация решения
        self._optimize_final_solution()

        objective = self._calculate_optimized_objective(self.assignment_matrix)

        self._analyze_optimized_performance(total_qubo_time, total_quantum_time, total_time,
                                            total_assigned, quantum_success_count, total_batches)

        return self.assignment_matrix, objective, total_time, total_qubo_time, total_quantum_time

    def _calculate_batch_complexity(self, batch_indices, top_servers):
        """Оценка сложности батча"""
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        total_load = batch_loads.sum()
        available_capacity = self.remaining_capacity[top_servers].sum()

        # Сложность = насколько близка нагрузка к доступной емкости
        complexity = min(1.0, total_load / (available_capacity + 1e-9))
        return complexity

    def _select_servers_optimized(self, batch_indices):
        """Оптимизированный выбор серверов"""
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        min_load = np.min(batch_loads)
        avg_load = np.mean(batch_loads)

        # Выбираем серверы с достаточной емкостью для средней нагрузки
        valid_servers = np.where(self.remaining_capacity >= avg_load * 0.7)[0]

        if len(valid_servers) == 0:
            # Если нет подходящих, берем с минимальной нагрузкой
            valid_servers = np.where(self.remaining_capacity >= min_load)[0]

        if len(valid_servers) == 0:
            return np.array([])

        scores = []
        for server_idx in valid_servers:
            server_capacity = self.remaining_capacity[server_idx]

            # Средняя стоимость для камер батча
            avg_cost = np.mean(self.cost_matrix[batch_indices, server_idx])

            # Количество камер, которые могут быть размещены
            feasible_cameras = np.sum(batch_loads <= server_capacity)

            capacity_score = server_capacity / np.max(self.remaining_capacity[valid_servers])
            cost_score = 1 - avg_cost
            feasibility_score = feasible_cameras / len(batch_indices)

            # Оптимизированная формула
            total_score = capacity_score * 0.3 + cost_score * 0.5 + feasibility_score * 0.2
            scores.append(total_score)

        scores = np.array(scores)
        top_indices = np.argsort(-scores)[:self.max_servers_per_batch]

        return valid_servers[top_indices]

    def _solve_batch_optimized(self, batch_indices, top_servers, num_reads):
        """Оптимизированное решение батча"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        qubo_start = time.time()
        Q = self._build_optimized_qubo(batch_indices, top_servers)
        qubo_time = time.time() - qubo_start

        if not Q:
            solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
            return solution, qubo_time, 0, False

        try:
            quantum_start = time.time()
            sampler = SimulatedAnnealingSampler()
            response = sampler.sample_qubo(Q, num_reads=num_reads)
            quantum_time = time.time() - quantum_start

            assignment = self._decode_optimized_solution(response, n_batch, top_servers, batch_indices)

            # Более строгая проверка качества
            valid_assignments = np.sum(assignment)
            if valid_assignments >= n_batch * 0.5:  # 50% успешных назначений
                return assignment, qubo_time, quantum_time, True
            else:
                solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
                return solution, qubo_time, quantum_time, False

        except Exception as e:
            logger.warning(f"Ошибка отжига: {e}, используем жадный алгоритм")
            solution = self._solve_batch_greedy_optimized(batch_indices, top_servers)
            return solution, qubo_time, 0, False

    def _build_optimized_qubo(self, batch_indices, top_servers):
        """ОПТИМИЗИРОВАННАЯ QUBO матрица"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        batch_costs = self.cost_matrix[batch_indices][:, top_servers]
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        batch_priorities = self.cameras_df.iloc[batch_indices]['priority'].values

        Q = {}

        # УЛУЧШЕННЫЕ НАГРАДЫ: более агрессивные за хорошие назначения
        for i in range(n_batch):
            priority_weight = (4 - batch_priorities[i])  # 3->1, 2->2, 1->3
            for j in range(n_servers_batch):
                var_name = f"x_{i}_{top_servers[j]}"
                cost = batch_costs[i, j]
                load = batch_loads[i]
                server_capacity = self.remaining_capacity[top_servers[j]]

                if load <= server_capacity:
                    # БОЛЬШАЯ награда за хорошие назначения
                    reward = -25.0 * (1.0 - cost) * priority_weight
                    Q[(var_name, var_name)] = reward
                else:
                    # БОЛЬШОЙ штраф за невозможные назначения
                    penalty = 100.0
                    Q[(var_name, var_name)] = penalty

        # БОЛЕЕ СИЛЬНЫЕ ограничения против конфликтов
        for i in range(n_batch):
            for j1 in range(n_servers_batch):
                var1 = f"x_{i}_{top_servers[j1]}"
                for j2 in range(j1 + 1, n_servers_batch):
                    var2 = f"x_{i}_{top_servers[j2]}"
                    Q[(var1, var2)] = 15.0  # Увеличенный штраф

        return Q

    def _decode_optimized_solution(self, response, n_batch, top_servers, batch_indices):
        """Оптимизированное декодирование"""
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        best_solution = response.first.sample

        # УЛУЧШЕННЫЙ алгоритм: учитываем стоимость и емкость
        candidates = []
        for i in range(n_batch):
            for j, server_idx in enumerate(top_servers):
                var_name = f"x_{i}_{server_idx}"
                if best_solution.get(var_name, 0) == 1:
                    cost = self.cost_matrix[batch_indices[i], server_idx]
                    priority = self.cameras_df.iloc[batch_indices[i]]['priority']
                    load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']
                    server_capacity = self.remaining_capacity[server_idx]

                    # Score: выше приоритет, ниже стоимость, лучше использование емкости
                    capacity_utilization = 1.0 - (load / server_capacity) if server_capacity > 0 else 0
                    score = priority * 0.4 + (1 - cost) * 0.4 + capacity_utilization * 0.2
                    candidates.append((i, j, score, cost, priority))

        # Сортируем по score
        candidates.sort(key=lambda x: -x[2])

        used_cameras = set()
        server_loads = {server_idx: 0 for server_idx in top_servers}

        for i, j, score, cost, priority in candidates:
            if i not in used_cameras:
                server_idx = top_servers[j]
                cam_idx = batch_indices[i]
                cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

                if server_loads[server_idx] + cam_load <= self.remaining_capacity[server_idx]:
                    assignment[i, j] = 1
                    used_cameras.add(i)
                    server_loads[server_idx] += cam_load

        return assignment

    def _post_process_solution(self, assignment, batch_indices, top_servers):
        """Пост-обработка решения для улучшения качества"""
        improved = assignment.copy()
        n_batch = len(batch_indices)

        # Локальный поиск: пробуем улучшить каждое назначение
        for i in range(n_batch):
            current_server = None
            current_cost = float('inf')

            # Находим текущее назначение
            for j in range(len(top_servers)):
                if assignment[i, j] == 1:
                    current_server = j
                    current_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                    break

            if current_server is not None:
                # Ищем лучшее назначение
                for j in range(len(top_servers)):
                    if j != current_server:
                        new_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                        cam_load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']

                        # Проверяем улучшение и допустимость
                        if (new_cost < current_cost * 0.95 and
                                self.remaining_capacity[top_servers[j]] >= cam_load):
                            improved[i, current_server] = 0
                            improved[i, j] = 1
                            break

        return improved

    def _optimize_final_solution(self):
        """Финальная оптимизация всего решения"""
        logger.info("Применение финальной оптимизации...")

        # Перераспределяем камеры для уменьшения стоимости
        for iteration in range(3):  # 3 итерации оптимизации
            improvements = 0
            for i in range(self.n_cameras):
                current_server = np.argmax(self.assignment_matrix[i]) if np.any(self.assignment_matrix[i]) else -1
                if current_server == -1:
                    continue

                current_cost = self.cost_matrix[i, current_server]
                current_load = self.cameras_df.iloc[i]['load_GFLOPS']

                # Ищем лучший сервер
                best_server = current_server
                best_cost = current_cost

                for j in range(self.n_servers):
                    if j != current_server:
                        new_cost = self.cost_matrix[i, j]
                        if (new_cost < best_cost * 0.98 and  # Значительное улучшение
                                self.remaining_capacity[j] >= current_load):

                            # Проверяем, что перераспределение возможно
                            if self._can_reassign(i, current_server, j):
                                best_server = j
                                best_cost = new_cost

                if best_server != current_server:
                    # Выполняем перераспределение
                    self.assignment_matrix[i, current_server] = 0
                    self.assignment_matrix[i, best_server] = 1
                    self.remaining_capacity[current_server] += current_load
                    self.remaining_capacity[best_server] -= current_load
                    improvements += 1

            if improvements == 0:
                break

            logger.info(f"Итерация {iteration + 1}: {improvements} улучшений")

    def _can_reassign(self, cam_idx, from_server, to_server):
        """Проверка возможности перераспределения"""
        cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
        return self.remaining_capacity[to_server] >= cam_load

    def _solve_batch_greedy_optimized(self, batch_indices, top_servers):
        """Оптимизированный жадный алгоритм"""
        n_batch = len(batch_indices)
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)

        remaining_capacity = self.remaining_capacity[top_servers].copy()

        batch_data = self.cameras_df.iloc[batch_indices]
        priority_scores = batch_data['priority'].values * batch_data['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for pos in sorted_indices:
            cam_idx = batch_indices[pos]
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j, server_idx in enumerate(top_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, server_idx]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[server_idx]['capacity_GFLOPS']

                    # Score: выше приоритет, ниже стоимость, лучше баланс нагрузки
                    score = (cam_priority * 0.5) + ((1 - cost) * 0.3) + (capacity_ratio * 0.2)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[pos, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        return assignment

    def solve_with_greedy_optimized(self):
        """Оптимизированный жадный алгоритм"""
        logger.info("=== ОПТИМИЗИРОВАННЫЙ ЖАДНЫЙ АЛГОРИТМ ===")
        start_time = time.time()

        assignment = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for cam_idx in sorted_indices:
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            cam_priority = self.cameras_df.iloc[cam_idx]['priority']

            best_server = -1
            best_score = -float('inf')

            for j in range(self.n_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, j]
                    capacity_ratio = remaining_capacity[j] / self.servers_df.iloc[j]['capacity_GFLOPS']

                    score = (cam_priority * 0.6) + ((1 - cost) * 0.4) + (capacity_ratio * 0.1)

                    if score > best_score:
                        best_score = score
                        best_server = j

            if best_server != -1:
                assignment[cam_idx, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        time_taken = time.time() - start_time
        objective = self._calculate_optimized_objective(assignment)
        covered = np.sum(np.any(assignment, axis=1))

        logger.info(f"Жадный алгоритм: {time_taken:.3f}с, {covered} камер")

        return assignment, objective, time_taken

    def _calculate_optimized_objective(self, assignment):
        """Оптимизированная целевая функция"""
        total_cost = 0
        uncovered_penalty = 0
        overload_penalty = 0


        for i in range(self.n_cameras):
            for j in range(self.n_servers):
                if assignment[i, j] == 1:
                    priority = self.cameras_df.iloc[i]['priority']
                    priority_weight = (4 - priority)
                    total_cost += self.cost_matrix[i, j] * priority_weight

        uncovered = self.n_cameras - np.sum(np.any(assignment, axis=1))
        uncovered_penalty = uncovered * 15.0

        for j in range(self.n_servers):
            server_load = np.sum(self.cameras_df['load_GFLOPS'].values * assignment[:, j])
            overload = max(0, server_load - self.servers_df.iloc[j]['capacity_GFLOPS'])
            overload_penalty += overload * 8.0

        total_objective = total_cost + uncovered_penalty + overload_penalty

        logger.info(
            f"Objective: cost={total_cost:.1f}, uncovered={uncovered_penalty:.1f}, overload={overload_penalty:.1f}")

        return total_objective

        logger.info("=" * 50)

        coverage = total_assigned / self.n_cameras * 100

        logger.info(f"Время QUBO: {total_qubo_time:.2f}с ({total_qubo_time / total_time * 100:.1f}%)")
        logger.info(f"Время отжига: {total_quantum_time:.2f}с ({quantum_efficiency:.1f}%)")

except ImportError:
    HAVE_DIMOD = False

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('quantum_fixed.log', encoding='utf-8'),

logger = logging.getLogger(__name__)


class FixedQuantumScheduler:
    def __init__(self, n_cameras=10000, n_servers=500, 
                 alpha=0.5, beta=0.3, gamma=0.2, delta=0.1,
                 A=100.0, B=200.0, C=0.1,  # ПЕРЕСМОТРЕННЫЕ КОЭФФИЦИЕНТЫ
                 batch_size=50, max_servers_per_batch=15,  # УМЕНЬШЕН БАТЧ
                 random_seed=42):

        logger.info(f"Инициализация: {n_cameras} камер, {n_servers} серверов")

        self.n_cameras = n_cameras
        self.n_servers = n_servers
        self.batch_size = batch_size
        self.max_servers_per_batch = max_servers_per_batch

        self.alpha = alpha
        self.beta = beta
        self.gamma = gamma
        self.delta = delta
        self.A = A  # Штраф за множественные назначения
        self.B = B  # Штраф за перегрузку
        self.C = C  # Вес стоимости

        self.random_seed = random_seed
        np.random.seed(self.random_seed)

        self.cameras_df = None
        self.servers_df = None
        self.cost_matrix = None
        self.assignment_matrix = None
        self.remaining_capacity = None
        self.assigned_cameras = set()

    def generate_realistic_data(self):
        """Генерация реалистичных данных"""
        logger.info("=== ГЕНЕРАЦИЯ ДАННЫХ ===")
        start_time = time.time()

        # Генерация камер (упрощенная версия)
        priorities = np.random.choice([3, 2, 1], size=self.n_cameras, p=[0.15, 0.25, 0.6])
        weights = np.zeros(self.n_cameras)
        loads = np.zeros(self.n_cameras)

        mask_p1 = priorities == 3
        mask_p2 = priorities == 2
        mask_p3 = priorities == 1

        weights[mask_p1] = np.random.uniform(4, 7, np.sum(mask_p1))
        loads[mask_p1] = np.random.uniform(8, 15, np.sum(mask_p1))

        weights[mask_p2] = np.random.uniform(2, 4, np.sum(mask_p2))
        loads[mask_p2] = np.random.uniform(4, 8, np.sum(mask_p2))

        weights[mask_p3] = np.random.uniform(0.4, 0.8, np.sum(mask_p3))
        loads[mask_p3] = np.random.uniform(1, 3, np.sum(mask_p3))

        camera_x = np.random.uniform(0, 1000, self.n_cameras)
        camera_y = np.random.uniform(0, 1000, self.n_cameras)

        self.cameras_df = pd.DataFrame({
            'camera_id': range(self.n_cameras),
            'priority': priorities,
            'weight_Mbps': weights,
            'load_GFLOPS': loads,
            'x': camera_x,
            'y': camera_y
        })

        # Генерация серверов
        server_x = np.random.uniform(0, 1000, self.n_servers)
        server_y = np.random.uniform(0, 1000, self.n_servers)
        
        server_types = np.random.choice([3, 2, 1], size=self.n_servers, p=[0.1, 0.3, 0.6])
        capacities = np.zeros(self.n_servers)

        capacities[server_types == 3] = np.random.uniform(800, 1000, np.sum(server_types == 3))
        capacities[server_types == 2] = np.random.uniform(400, 800, np.sum(server_types == 2))
        capacities[server_types == 1] = np.random.uniform(200, 400, np.sum(server_types == 1))

        self.servers_df = pd.DataFrame({
            'server_id': range(self.n_servers),
            'capacity_GFLOPS': capacities,
            'x': server_x,
            'y': server_y
        })

        # Матрица стоимости
        self._build_cost_matrix()
        self.remaining_capacity = capacities.copy()

        gen_time = time.time() - start_time
        logger.info(f"Данные сгенерированы за {gen_time:.2f}с")
        logger.info(f"Общая нагрузка камер: {loads.sum():.1f}, емкость серверов: {capacities.sum():.1f}")

    def _build_cost_matrix(self):
        """Построение матрицы стоимости"""
        cam_coords = self.cameras_df[['x', 'y']].values
        srv_coords = self.servers_df[['x', 'y']].values

        distances = distance_matrix(cam_coords, srv_coords)
        distances_norm = distances / np.max(distances)

        priorities_norm = (3 - self.cameras_df['priority'].values) / 2.0  # 0 to 1
        loads_norm = self.cameras_df['load_GFLOPS'].values / np.max(self.cameras_df['load_GFLOPS'])

        capacities_inv = 1 / (self.servers_df['capacity_GFLOPS'].values + 1e-9)
        capacities_norm = capacities_inv / np.max(capacities_inv)

        self.cost_matrix = (
            self.alpha * distances_norm +
            self.beta * loads_norm[:, None] +
            self.gamma * priorities_norm[:, None] +
            self.delta * capacities_norm[None, :]
        )

        self.cost_matrix = self.cost_matrix / np.max(self.cost_matrix)

    def solve_with_quantum_annealing_optimized(self, num_reads=50):
        """Оптимизированный квантовый отжиг"""
        if not HAVE_DIMOD:
            logger.error("Dimod недоступен")
            return None, 0, 0, 0, 0

        logger.info("=== ОПТИМИЗИРОВАННЫЙ КВАНТОВЫЙ ОТЖИГ ===")
        total_start = time.time()

        self.assignment_matrix = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        self.remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()
        self.assigned_cameras = set()

        # Сортируем по приоритету
        sorted_indices = np.argsort(-self.cameras_df['priority'].values)
        total_batches = (self.n_cameras + self.batch_size - 1) // self.batch_size

        total_qubo_time = 0
        total_quantum_time = 0
        quantum_success_count = 0
        total_assigned = 0

        logger.info(f"Обработка {total_batches} батчей...")

        for batch_idx in range(total_batches):
            # Ранний выход
            if len(self.assigned_cameras) / self.n_cameras > 0.95:
                logger.info(f"Достигнуто 95% покрытие, завершаем")
                break

            start_idx = batch_idx * self.batch_size
            end_idx = min((batch_idx + 1) * self.batch_size, self.n_cameras)
            batch_indices = sorted_indices[start_idx:end_idx]

            available_indices = [idx for idx in batch_indices if idx not in self.assigned_cameras]
            if not available_indices:
                continue

            # Выбор серверов с достаточной емкостью
            batch_load = self.cameras_df.iloc[available_indices]['load_GFLOPS'].sum()
            top_servers = self._select_best_servers(available_indices, batch_load)

            if len(top_servers) == 0:
                # Если нет подходящих серверов, используем жадный
                greedy_solution = self._solve_batch_greedy(available_indices, list(range(self.n_servers)))
                for i, cam_idx in enumerate(available_indices):
                    for j in range(self.n_servers):
                        if greedy_solution[i, j] == 1:
                            self.assignment_matrix[cam_idx, j] = 1
                            self.remaining_capacity[j] -= self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                            self.assigned_cameras.add(cam_idx)
                            total_assigned += 1
                continue

            # Решение батча
            batch_solution, qubo_time, quantum_time, quantum_used = self._solve_batch_optimized(
                available_indices, top_servers, num_reads
            )

            total_qubo_time += qubo_time
            total_quantum_time += quantum_time
            if quantum_used:
                quantum_success_count += 1

            # Обновление назначений
            batch_assigned = 0
            for i, cam_idx in enumerate(available_indices):
                for j_local, j_global in enumerate(top_servers):
                    if batch_solution[i, j_local] == 1:
                        self.assignment_matrix[cam_idx, j_global] = 1
                        self.remaining_capacity[j_global] -= self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                        self.assigned_cameras.add(cam_idx)
                        batch_assigned += 1
                        break  # Только один сервер на камеру

            total_assigned += batch_assigned

            if (batch_idx + 1) % 20 == 0:
                logger.info(f"Батч {batch_idx + 1}: {batch_assigned} камер")

        total_time = time.time() - total_start
        objective = self._calculate_objective_simple(self.assignment_matrix)

        self._analyze_performance(total_qubo_time, total_quantum_time, total_time,
                                 total_assigned, quantum_success_count, total_batches)

        return self.assignment_matrix, objective, total_time, total_qubo_time, total_quantum_time

    def _select_best_servers(self, batch_indices, batch_load):
        """Выбор лучших серверов для батча"""
        # Серверы с достаточной остаточной емкостью
        valid_servers = np.where(self.remaining_capacity >= self.cameras_df.iloc[batch_indices]['load_GFLOPS'].min())[0]
        
        if len(valid_servers) == 0:
            return np.array([])
        
        # Сортируем по емкости и стоимости
        batch_costs = self.cost_matrix[batch_indices][:, valid_servers].mean(axis=0)
        capacities = self.remaining_capacity[valid_servers]
        
        # Комбинированный score
        cost_norm = (batch_costs - batch_costs.min()) / (batch_costs.max() - batch_costs.min() + 1e-9)
        cap_norm = (capacities - capacities.min()) / (capacities.max() - capacities.min() + 1e-9)
        
        scores = cap_norm * 0.7 - cost_norm * 0.3
        top_indices = np.argsort(-scores)[:self.max_servers_per_batch]
        
        return valid_servers[top_indices]

    def _solve_batch_optimized(self, batch_indices, top_servers, num_reads):
        """Оптимизированное решение батча"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        # Построение QUBO
        qubo_start = time.time()
        Q = self._build_simple_qubo(batch_indices, top_servers)
        qubo_time = time.time() - qubo_start

        if not Q or len(Q) == 0:
            return self._solve_batch_greedy(batch_indices, top_servers), qubo_time, 0, False

        try:
            # Запуск отжига
            quantum_start = time.time()
            sampler = SimulatedAnnealingSampler()
            response = sampler.sample_qubo(Q, num_reads=num_reads)
            quantum_time = time.time() - quantum_start

            # Декодирование решения
            assignment = self._decode_solution(response, n_batch, top_servers)
            
            # Проверка допустимости решения
            if self._is_solution_valid(assignment, batch_indices, top_servers):
                return assignment, qubo_time, quantum_time, True
            else:
                return self._solve_batch_greedy(batch_indices, top_servers), qubo_time, quantum_time, False

        except Exception as e:
            logger.warning(f"Ошибка отжига: {e}, используем жадный алгоритм")
            return self._solve_batch_greedy(batch_indices, top_servers), qubo_time, 0, False

    def _build_simple_qubo(self, batch_indices, top_servers):
        """ПРОСТАЯ И ПРАВИЛЬНАЯ QUBO матрица"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)
        
        batch_cameras = self.cameras_df.iloc[batch_indices]
        batch_costs = self.cost_matrix[batch_indices][:, top_servers]
        batch_loads = batch_cameras['load_GFLOPS'].values
        batch_priorities = batch_cameras['priority'].values
        
        Q = {}
        
        # 1. ЦЕЛЕВАЯ ФУНКЦИЯ: минимизация стоимости
        for i in range(n_batch):
            priority_weight = (4 - batch_priorities[i])  # 3->1, 2->2, 1->3
            for j in range(n_servers_batch):
                var_name = f"x_{i}_{top_servers[j]}"
                cost = batch_costs[i, j] * priority_weight
                Q[(var_name, var_name)] = self.C * cost
        
        # 2. ОГРАНИЧЕНИЕ: одна камера - один сервер
        for i in range(n_batch):
            for j1 in range(n_servers_batch):
                var1 = f"x_{i}_{top_servers[j1]}"
                for j2 in range(j1 + 1, n_servers_batch):
                    var2 = f"x_{i}_{top_servers[j2]}"
                    Q[(var1, var2)] = self.A
        
        # 3. ОГРАНИЧЕНИЕ: не превышать емкость серверов
        for j in range(n_servers_batch):
            server_idx = top_servers[j]
            server_capacity = self.remaining_capacity[server_idx]
            
            # Для каждой пары камер, которые вместе перегружают сервер
            for i1 in range(n_batch):
                load1 = batch_loads[i1]
                var1 = f"x_{i1}_{server_idx}"
                
                for i2 in range(i1 + 1, n_batch):
                    load2 = batch_loads[i2]
                    if load1 + load2 > server_capacity:
                        var2 = f"x_{i2}_{server_idx}"
                        Q[(var1, var2)] = Q.get((var1, var2), 0) + self.B
        
        return Q

    def _decode_solution(self, response, n_batch, top_servers):
        """Декодирование решения из сэмплера"""
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        best_solution = response.first.sample
        
        for i in range(n_batch):
            # Находим сервер с минимальной стоимостью среди выбранных
            best_server = -1
            best_cost = float('inf')
            
            for j, server_idx in enumerate(top_servers):
                var_name = f"x_{i}_{server_idx}"
                if best_solution.get(var_name, 0) == 1:
                    cost = self.cost_matrix[i, server_idx] if i < self.n_cameras else float('inf')
                    if cost < best_cost:
                        best_cost = cost
                        best_server = j
            
            if best_server != -1:
                assignment[i, best_server] = 1
        
        return assignment

    def _is_solution_valid(self, assignment, batch_indices, top_servers):
        """Проверка допустимости решения"""
        # Проверяем, что камеры назначены не более чем на один сервер
        if np.any(assignment.sum(axis=1) > 1):
            return False
        
        # Проверяем перегрузку серверов
        for j, server_idx in enumerate(top_servers):
            total_load = 0
            for i, cam_idx in enumerate(batch_indices):
                if assignment[i, j] == 1:
                    total_load += self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            if total_load > self.remaining_capacity[server_idx]:
                return False
        
        return True

    def _solve_batch_greedy(self, batch_indices, top_servers):
        """Жадный алгоритм для батча"""
        n_batch = len(batch_indices)
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        
        remaining_capacity = self.remaining_capacity[top_servers].copy()
        batch_priorities = self.cameras_df.iloc[batch_indices]['priority'].values
        
        # Сортируем по приоритету
        sorted_indices = np.argsort(-batch_priorities)
        
        for pos in sorted_indices:
            cam_idx = batch_indices[pos]
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            
            best_server = -1
            best_cost = float('inf')
            
            for j, server_idx in enumerate(top_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, server_idx]
                    if cost < best_cost:
                        best_cost = cost
                        best_server = j
            
            if best_server != -1:
                assignment[pos, best_server] = 1
                remaining_capacity[best_server] -= cam_load
        
        return assignment

    def solve_with_greedy(self):
        """Классический жадный алгоритм"""
        logger.info("=== ЖАДНЫЙ АЛГОРИТМ ===")
        start_time = time.time()

        assignment = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()

        # Сортируем камеры по приоритету
        sorted_indices = np.argsort(-self.cameras_df['priority'].values)

        for cam_idx in sorted_indices:
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
            
            best_server = -1
            best_cost = float('inf')
            
            for j in range(self.n_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, j]
                    if cost < best_cost:
                        best_cost = cost
                        best_server = j
            
            if best_server != -1:
                assignment[cam_idx, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        time_taken = time.time() - start_time
        objective = self._calculate_objective_simple(assignment)
        covered = np.sum(np.any(assignment, axis=1))

        logger.info(f"Жадный алгоритм: {time_taken:.3f}с, {covered} камер")

        return assignment, objective, time_taken

    def _calculate_objective_simple(self, assignment):
        """УПРОЩЕННАЯ целевая функция"""
        total_cost = 0
        uncovered = 0
        overload_penalty = 0

        # Стоимость назначений
        for i in range(self.n_cameras):
            for j in range(self.n_servers):
                if assignment[i, j] == 1:
                    priority = self.cameras_df.iloc[i]['priority']
                    priority_weight = (4 - priority)  # Высокий приоритет = больше вес
                    total_cost += self.cost_matrix[i, j] * priority_weight * self.C

        # Непокрытые камеры
        uncovered = self.n_cameras - np.sum(np.any(assignment, axis=1))
        uncovered_penalty = uncovered * self.A * 100

        # Перегрузка серверов
        for j in range(self.n_servers):
            server_load = np.sum(self.cameras_df['load_GFLOPS'].values * assignment[:, j])
            overload = max(0, server_load - self.servers_df.iloc[j]['capacity_GFLOPS'])
            overload_penalty += overload * self.B * 10

        total_objective = total_cost + uncovered_penalty + overload_penalty
        
        logger.debug(f"Objective breakdown: cost={total_cost:.1f}, uncovered_penalty={uncovered_penalty:.1f}, overload_penalty={overload_penalty:.1f}")
        
        return total_objective

    def _analyze_performance(self, total_qubo_time, total_quantum_time, total_time,
                            total_assigned, quantum_success, total_batches):
        """Анализ производительности"""
        logger.info("\n" + "=" * 50)
        logger.info("АНАЛИЗ ПРОИЗВОДИТЕЛЬНОСТИ")
        logger.info("=" * 50)

        coverage = total_assigned / self.n_cameras * 100
        quantum_efficiency = (total_quantum_time / total_time * 100) if total_time > 0 else 0

        logger.info(f"Успешных QUBO батчей: {quantum_success}/{total_batches} ({quantum_success/total_batches*100:.1f}%)")
        logger.info(f"Покрытие: {total_assigned}/{self.n_cameras} ({coverage:.1f}%)")
        logger.info(f"Общее время: {total_time:.2f}с")
        logger.info(f"Время QUBO: {total_qubo_time:.2f}с ({total_qubo_time/total_time*100:.1f}%)")
        logger.info(f"Время отжига: {total_quantum_time:.2f}с ({quantum_efficiency:.1f}%)")

    def run_comparison(self):
        """Сравнение методов"""
        logger.info("=== СРАВНЕНИЕ МЕТОДОВ ===")

        results = {}

        # 1. Жадный алгоритм
        logger.info("\n1. Запуск жадного алгоритма...")
        greedy_assignment, greedy_objective, greedy_time = self.solve_with_greedy()
        results['Greedy'] = {
            'assignment': greedy_assignment,
            'objective': greedy_objective,
            'time': greedy_time,
            'covered': np.sum(np.any(greedy_assignment, axis=1))
        }

        # 2. Квантовый отжиг
        if HAVE_DIMOD:
            logger.info("\n2. Запуск квантового отжига...")
            quantum_assignment, quantum_objective, quantum_time, quantum_qubo, quantum_annealing = \
                self.solve_with_quantum_annealing_optimized(num_reads=30)

            results['Quantum-Optimized'] = {
                'assignment': quantum_assignment,
                'objective': quantum_objective,
                'time': quantum_time,
                'qubo_time': quantum_qubo,
                'quantum_time': quantum_annealing,
                'covered': np.sum(np.any(quantum_assignment, axis=1))
            }

        # Вывод результатов
        self._print_results(results)
        return results

    def _print_results(self, results):

class BalancedQuantumScheduler:
    def __init__(self, n_cameras=5000, n_servers=250,
                 batch_size=40, max_servers_per_batch=15,  # Возвращаем рабочие параметры
                 random_seed=42):

        logger.info(f"Инициализация: {n_cameras} камер, {n_servers} серверов")

        self.n_cameras = n_cameras
        self.n_servers = n_servers
        self.batch_size = batch_size
        self.max_servers_per_batch = max_servers_per_batch

        self.random_seed = random_seed
        np.random.seed(self.random_seed)

        self.cameras_df = None
        self.servers_df = None
        self.cost_matrix = None
        self.assignment_matrix = None
        self.remaining_capacity = None
        self.assigned_cameras = set()

    def generate_realistic_data(self):
        """Генерация данных"""
        logger.info("=== ГЕНЕРАЦИЯ ДАННЫХ ===")

        priorities = np.random.choice([3, 2, 1], size=self.n_cameras, p=[0.15, 0.25, 0.6])
        weights = np.zeros(self.n_cameras)
        loads = np.zeros(self.n_cameras)

        mask_p1 = priorities == 3
        mask_p2 = priorities == 2
        mask_p3 = priorities == 1

        weights[mask_p1] = np.random.uniform(4, 7, np.sum(mask_p1))
        loads[mask_p1] = np.random.uniform(8, 15, np.sum(mask_p1))

        weights[mask_p2] = np.random.uniform(2, 4, np.sum(mask_p2))
        loads[mask_p2] = np.random.uniform(4, 8, np.sum(mask_p2))

        weights[mask_p3] = np.random.uniform(0.4, 0.8, np.sum(mask_p3))
        loads[mask_p3] = np.random.uniform(1, 3, np.sum(mask_p3))

        camera_x = np.random.uniform(0, 1000, self.n_cameras)
        camera_y = np.random.uniform(0, 1000, self.n_cameras)

        self.cameras_df = pd.DataFrame({
            'camera_id': range(self.n_cameras),
            'priority': priorities,
            'weight_Mbps': weights,
            'load_GFLOPS': loads,
            'x': camera_x,
            'y': camera_y
        })

        server_x = np.random.uniform(0, 1000, self.n_servers)
        server_y = np.random.uniform(0, 1000, self.n_servers)

        server_types = np.random.choice([3, 2, 1], size=self.n_servers, p=[0.1, 0.3, 0.6])
        capacities = np.zeros(self.n_servers)

        capacities[server_types == 3] = np.random.uniform(800, 1000, np.sum(server_types == 3))
        capacities[server_types == 2] = np.random.uniform(400, 800, np.sum(server_types == 2))
        capacities[server_types == 1] = np.random.uniform(200, 400, np.sum(server_types == 1))

        self.servers_df = pd.DataFrame({
            'server_id': range(self.n_servers),
            'capacity_GFLOPS': capacities,
            'x': server_x,
            'y': server_y
        })

        self._build_balanced_cost_matrix()
        self.remaining_capacity = capacities.copy()

        total_load = loads.sum()
        total_capacity = capacities.sum()
        utilization = total_load / total_capacity * 100
        logger.info(
            f"Общая нагрузка: {total_load:.1f}, емкость: {total_capacity:.1f}, использование: {utilization:.1f}%")

        return utilization

    def _build_balanced_cost_matrix(self):
        """Сбалансированная матрица стоимости"""
        cam_coords = self.cameras_df[['x', 'y']].values
        srv_coords = self.servers_df[['x', 'y']].values

        distances = distance_matrix(cam_coords, srv_coords)
        distances_norm = distances / np.max(distances)

        priorities_norm = (3 - self.cameras_df['priority'].values) / 2.0
        loads_norm = self.cameras_df['load_GFLOPS'].values / np.max(self.cameras_df['load_GFLOPS'])

        capacities_inv = 1 / (self.servers_df['capacity_GFLOPS'].values + 1e-9)
        capacities_norm = capacities_inv / np.max(capacities_inv)

        # СБАЛАНСИРОВАННЫЕ веса
        self.cost_matrix = (
                0.5 * distances_norm +  # Умеренный вес расстояниям
                0.2 * loads_norm[:, None] +  # Умеренный вес нагрузке
                0.2 * priorities_norm[:, None] +  # Умеренный вес приоритету
                0.1 * capacities_norm[None, :]  # Небольшой вес емкости
        )

        self.cost_matrix = (self.cost_matrix - np.min(self.cost_matrix)) / (
                    np.max(self.cost_matrix) - np.min(self.cost_matrix) + 1e-9)

    def solve_with_quantum_balanced(self, num_reads=100):
        """СБАЛАНСИРОВАННЫЙ квантовый отжиг"""
        if not HAVE_DIMOD:
            logger.error("Dimod недоступен")
            return None, 0, 0, 0, 0

        logger.info("=== СБАЛАНСИРОВАННЫЙ КВАНТОВЫЙ ОТЖИГ ===")
        total_start = time.time()

        self.assignment_matrix = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        self.remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()
        self.assigned_cameras = set()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)
        total_batches = (self.n_cameras + self.batch_size - 1) // self.batch_size

        total_qubo_time = 0
        total_quantum_time = 0
        quantum_success_count = 0
        total_assigned = 0

        logger.info(f"Обработка {total_batches} батчей...")

        for batch_idx in range(total_batches):
            # НОРМАЛЬНЫЙ критерий остановки
            if len(self.assigned_cameras) / self.n_cameras > 0.98:
                logger.info(f"Достигнуто 98% покрытие, завершаем")
                break

            start_idx = batch_idx * self.batch_size
            end_idx = min((batch_idx + 1) * self.batch_size, self.n_cameras)
            batch_indices = sorted_indices[start_idx:end_idx]

            available_indices = [idx for idx in batch_indices if idx not in self.assigned_cameras]
            if not available_indices:
                continue

            # НОРМАЛЬНЫЙ выбор серверов
            top_servers = self._select_servers_balanced(available_indices)
            if len(top_servers) == 0:
                # Если нет серверов, используем все доступные
                top_servers = list(range(min(20, self.n_servers)))
                if len(top_servers) == 0:
                    continue

            batch_solution, qubo_time, quantum_time, quantum_used = self._solve_batch_balanced(
                available_indices, top_servers, num_reads
            )

            total_qubo_time += qubo_time
            total_quantum_time += quantum_time
            if quantum_used:
                quantum_success_count += 1

            # УМЕРЕННАЯ пост-обработка
            batch_solution = self._balanced_post_process(batch_solution, available_indices, top_servers)

            batch_assigned = 0
            for i, cam_idx in enumerate(available_indices):
                assigned = False
                for j_local, j_global in enumerate(top_servers):
                    if batch_solution[i, j_local] == 1 and not assigned:
                        if self.remaining_capacity[j_global] >= self.cameras_df.iloc[cam_idx]['load_GFLOPS']:
                            self.assignment_matrix[cam_idx, j_global] = 1
                            self.remaining_capacity[j_global] -= self.cameras_df.iloc[cam_idx]['load_GFLOPS']
                            self.assigned_cameras.add(cam_idx)
                            batch_assigned += 1
                            assigned = True

            total_assigned += batch_assigned

            if (batch_idx + 1) % 20 == 0:
                current_coverage = len(self.assigned_cameras) / self.n_cameras * 100
                success_rate = quantum_success_count / (batch_idx + 1) * 100
                logger.info(
                    f"Батч {batch_idx + 1}: {batch_assigned} камер, покрытие: {current_coverage:.1f}%, успех QUBO: {success_rate:.1f}%")

        total_time = time.time() - total_start

        # УМЕРЕННАЯ финальная оптимизация
        self._balanced_optimization()

        objective = self._calculate_balanced_objective(self.assignment_matrix)

        self._analyze_balanced_performance(total_qubo_time, total_quantum_time, total_time,
                                           total_assigned, quantum_success_count, total_batches)

        return self.assignment_matrix, objective, total_time, total_qubo_time, total_quantum_time

    def _select_servers_balanced(self, batch_indices):
        """Сбалансированный выбор серверов"""
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        min_load = np.min(batch_loads)

        # ЛОЯЛЬНЫЙ отбор серверов
        valid_servers = np.where(self.remaining_capacity >= min_load)[0]

        if len(valid_servers) == 0:
            return np.array([])

        scores = []
        for server_idx in valid_servers:
            server_capacity = self.remaining_capacity[server_idx]
            avg_cost = np.mean(self.cost_matrix[batch_indices, server_idx])

            capacity_score = server_capacity / np.max(self.remaining_capacity[valid_servers])
            cost_score = 1 - avg_cost

            # СБАЛАНСИРОВАННАЯ формула
            total_score = capacity_score * 0.4 + cost_score * 0.6
            scores.append(total_score)

        scores = np.array(scores)
        top_indices = np.argsort(-scores)[:self.max_servers_per_batch]

        return valid_servers[top_indices]

    def _solve_batch_balanced(self, batch_indices, top_servers, num_reads):
        """Сбалансированное решение батча"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        qubo_start = time.time()
        Q = self._build_balanced_qubo(batch_indices, top_servers)
        qubo_time = time.time() - qubo_start

        if not Q:
            solution = self._solve_batch_greedy_balanced(batch_indices, top_servers)
            return solution, qubo_time, 0, False

        try:
            quantum_start = time.time()
            sampler = SimulatedAnnealingSampler()
            response = sampler.sample_qubo(Q, num_reads=num_reads)
            quantum_time = time.time() - quantum_start

            assignment = self._decode_balanced_solution(response, n_batch, top_servers, batch_indices)

            # ЛОЯЛЬНАЯ проверка качества
            valid_assignments = np.sum(assignment)
            if valid_assignments >= n_batch * 0.3:  # 30% успешных назначений
                return assignment, qubo_time, quantum_time, True
            else:
                solution = self._solve_batch_greedy_balanced(batch_indices, top_servers)
                return solution, qubo_time, quantum_time, False

        except Exception as e:
            logger.warning(f"Ошибка отжига: {e}, используем жадный алгоритм")
            solution = self._solve_batch_greedy_balanced(batch_indices, top_servers)
            return solution, qubo_time, 0, False

    def _build_balanced_qubo(self, batch_indices, top_servers):
        """СБАЛАНСИРОВАННАЯ QUBO матрица"""
        n_batch = len(batch_indices)
        n_servers_batch = len(top_servers)

        batch_costs = self.cost_matrix[batch_indices][:, top_servers]
        batch_loads = self.cameras_df.iloc[batch_indices]['load_GFLOPS'].values
        batch_priorities = self.cameras_df.iloc[batch_indices]['priority'].values

        Q = {}

        # СБАЛАНСИРОВАННЫЕ награды и штрафы
        for i in range(n_batch):
            priority_weight = (4 - batch_priorities[i])
            for j in range(n_servers_batch):
                var_name = f"x_{i}_{top_servers[j]}"
                cost = batch_costs[i, j]
                load = batch_loads[i]
                server_capacity = self.remaining_capacity[top_servers[j]]

                if load <= server_capacity:
                    # УМЕРЕННАЯ награда
                    reward = -20.0 * (1.0 - cost) * priority_weight
                    Q[(var_name, var_name)] = reward
                else:
                    # УМЕРЕННЫЙ штраф
                    penalty = 50.0
                    Q[(var_name, var_name)] = penalty

        # УМЕРЕННЫЕ ограничения
        for i in range(n_batch):
            for j1 in range(n_servers_batch):
                var1 = f"x_{i}_{top_servers[j1]}"
                for j2 in range(j1 + 1, n_servers_batch):
                    var2 = f"x_{i}_{top_servers[j2]}"
                    Q[(var1, var2)] = 10.0

        return Q

    def _decode_balanced_solution(self, response, n_batch, top_servers, batch_indices):
        """Сбалансированное декодирование"""
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)
        best_solution = response.first.sample

        candidates = []
        for i in range(n_batch):
            for j, server_idx in enumerate(top_servers):
                var_name = f"x_{i}_{server_idx}"
                if best_solution.get(var_name, 0) == 1:
                    cost = self.cost_matrix[batch_indices[i], server_idx]
                    priority = self.cameras_df.iloc[batch_indices[i]]['priority']
                    load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']
                    server_capacity = self.remaining_capacity[server_idx]

                    # СБАЛАНСИРОВАННЫЙ score
                    capacity_utilization = 1.0 - (load / server_capacity) if server_capacity > 0 else 0
                    score = priority * 0.3 + (1 - cost) * 0.5 + capacity_utilization * 0.2
                    candidates.append((i, j, score))

        candidates.sort(key=lambda x: -x[2])

        used_cameras = set()
        server_loads = {server_idx: 0 for server_idx in top_servers}

        for i, j, score in candidates:
            if i not in used_cameras:
                server_idx = top_servers[j]
                cam_idx = batch_indices[i]
                cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

                if server_loads[server_idx] + cam_load <= self.remaining_capacity[server_idx]:
                    assignment[i, j] = 1
                    used_cameras.add(i)
                    server_loads[server_idx] += cam_load

        return assignment

    def _balanced_post_process(self, assignment, batch_indices, top_servers):
        """Умеренная пост-обработка"""
        improved = assignment.copy()
        n_batch = len(batch_indices)

        for i in range(n_batch):
            current_server = None
            current_cost = float('inf')

            for j in range(len(top_servers)):
                if assignment[i, j] == 1:
                    current_server = j
                    current_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                    break

            if current_server is not None:
                for j in range(len(top_servers)):
                    if j != current_server:
                        new_cost = self.cost_matrix[batch_indices[i], top_servers[j]]
                        cam_load = self.cameras_df.iloc[batch_indices[i]]['load_GFLOPS']

                        if (new_cost < current_cost * 0.95 and  # 5% улучшение
                                self.remaining_capacity[top_servers[j]] >= cam_load):
                            improved[i, current_server] = 0
                            improved[i, j] = 1
                            break

        return improved

    def _balanced_optimization(self):
        """Умеренная финальная оптимизация"""
        logger.info("Применение умеренной оптимизации...")

        improvements = 0
        for i in range(self.n_cameras):
            current_server = np.argmax(self.assignment_matrix[i]) if np.any(self.assignment_matrix[i]) else -1
            if current_server == -1:
                continue

            current_cost = self.cost_matrix[i, current_server]
            current_load = self.cameras_df.iloc[i]['load_GFLOPS']

            best_server = current_server
            best_cost = current_cost

            for j in range(self.n_servers):
                if j != current_server:
                    new_cost = self.cost_matrix[i, j]

                    if (new_cost < best_cost and
                            self.remaining_capacity[j] >= current_load and
                            self._can_reassign(i, current_server, j)):
                        best_server = j
                        best_cost = new_cost

            if best_server != current_server:
                self.assignment_matrix[i, current_server] = 0
                self.assignment_matrix[i, best_server] = 1
                self.remaining_capacity[current_server] += current_load
                self.remaining_capacity[best_server] -= current_load
                improvements += 1

        logger.info(f"Оптимизация: {improvements} улучшений")

    def _can_reassign(self, cam_idx, from_server, to_server):
        """Проверка перераспределения"""
        cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']
        return self.remaining_capacity[to_server] >= cam_load

    def _solve_batch_greedy_balanced(self, batch_indices, top_servers):
        """Сбалансированный жадный алгоритм"""
        n_batch = len(batch_indices)
        assignment = np.zeros((n_batch, len(top_servers)), dtype=int)

        remaining_capacity = self.remaining_capacity[top_servers].copy()

        batch_data = self.cameras_df.iloc[batch_indices]
        priority_scores = batch_data['priority'].values * batch_data['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for pos in sorted_indices:
            cam_idx = batch_indices[pos]
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

            best_server = -1
            best_cost = float('inf')

            for j, server_idx in enumerate(top_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, server_idx]
                    if cost < best_cost:
                        best_cost = cost
                        best_server = j

            if best_server != -1:
                assignment[pos, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        return assignment

    def solve_with_greedy_balanced(self):
        """Сбалансированный жадный алгоритм"""
        logger.info("=== СБАЛАНСИРОВАННЫЙ ЖАДНЫЙ АЛГОРИТМ ===")
        start_time = time.time()

        assignment = np.zeros((self.n_cameras, self.n_servers), dtype=int)
        remaining_capacity = self.servers_df['capacity_GFLOPS'].values.copy()

        priority_scores = self.cameras_df['priority'].values * self.cameras_df['load_GFLOPS'].values
        sorted_indices = np.argsort(-priority_scores)

        for cam_idx in sorted_indices:
            cam_load = self.cameras_df.iloc[cam_idx]['load_GFLOPS']

            best_server = -1
            best_cost = float('inf')

            for j in range(self.n_servers):
                if remaining_capacity[j] >= cam_load:
                    cost = self.cost_matrix[cam_idx, j]
                    if cost < best_cost:
                        best_cost = cost
                        best_server = j

            if best_server != -1:
                assignment[cam_idx, best_server] = 1
                remaining_capacity[best_server] -= cam_load

        time_taken = time.time() - start_time
        objective = self._calculate_balanced_objective(assignment)
        covered = np.sum(np.any(assignment, axis=1))

        logger.info(f"Жадный алгоритм: {time_taken:.3f}с, {covered} камер")

        return assignment, objective, time_taken

    def _calculate_balanced_objective(self, assignment):
        """Сбалансированная целевая функция"""
        total_cost = 0
        uncovered_penalty = 0
        overload_penalty = 0

        for i in range(self.n_cameras):
            for j in range(self.n_servers):
                if assignment[i, j] == 1:
                    priority = self.cameras_df.iloc[i]['priority']
                    priority_weight = (4 - priority)
                    total_cost += self.cost_matrix[i, j] * priority_weight

        uncovered = self.n_cameras - np.sum(np.any(assignment, axis=1))
        uncovered_penalty = uncovered * 10.0  # УМЕРЕННЫЙ штраф

        for j in range(self.n_servers):
            server_load = np.sum(self.cameras_df['load_GFLOPS'].values * assignment[:, j])
            overload = max(0, server_load - self.servers_df.iloc[j]['capacity_GFLOPS'])
            overload_penalty += overload * 5.0  # УМЕРЕННЫЙ штраф

        total_objective = total_cost + uncovered_penalty + overload_penalty

        logger.info(
            f"Objective: cost={total_cost:.1f}, uncovered={uncovered_penalty:.1f}, overload={overload_penalty:.1f}")

        return total_objective

    def _analyze_balanced_performance(self, total_qubo_time, total_quantum_time, total_time,
                                      total_assigned, quantum_success, total_batches):
        """Анализ производительности"""
        logger.info("\n" + "=" * 50)
        logger.info("АНАЛИЗ СБАЛАНСИРОВАННЫХ РЕЗУЛЬТАТОВ")
        logger.info("=" * 50)

        coverage = total_assigned / self.n_cameras * 100
        quantum_efficiency = (total_quantum_time / total_time * 100) if total_time > 0 else 0

        logger.info(
            f"Успешных QUBO батчей: {quantum_success}/{total_batches} ({quantum_success / total_batches * 100:.1f}%)")
        logger.info(f"Покрытие: {total_assigned}/{self.n_cameras} ({coverage:.1f}%)")
        logger.info(f"Общее время: {total_time:.2f}с")
        logger.info(f"Время QUBO: {total_qubo_time:.2f}с ({total_qubo_time / total_time * 100:.1f}%)")
        logger.info(f"Время отжига: {total_quantum_time:.2f}с ({quantum_efficiency:.1f}%)")

    def run_balanced_comparison(self):
        """Сбалансированное сравнение"""
        logger.info("=== СБАЛАНСИРОВАННОЕ СРАВНЕНИЕ ===")

        results = {}

        logger.info("\n1. Запуск сбалансированного жадного алгоритма...")
        greedy_assignment, greedy_objective, greedy_time = self.solve_with_greedy_balanced()
        results['Greedy-Balanced'] = {
            'assignment': greedy_assignment,
            'objective': greedy_objective,
            'time': greedy_time,
            'covered': np.sum(np.any(greedy_assignment, axis=1))
        }

        if HAVE_DIMOD:
            logger.info("\n2. Запуск сбалансированного квантового отжига...")
            quantum_assignment, quantum_objective, quantum_time, quantum_qubo, quantum_annealing = \
                self.solve_with_quantum_balanced(num_reads=100)

            results['Quantum-Balanced'] = {
                'assignment': quantum_assignment,
                'objective': quantum_objective,
                'time': quantum_time,
                'qubo_time': quantum_qubo,
                'quantum_time': quantum_annealing,
                'covered': np.sum(np.any(quantum_assignment, axis=1))
            }

        self._print_balanced_results(results)
        return results

    def _print_balanced_results(self, results):
        """Печать сбалансированных результатов"""
        print("\n" + "=" * 100)
        print("СБАЛАНСИРОВАННЫЕ РЕЗУЛЬТАТЫ")
        print("=" * 100)
        print(f"{'МЕТОД':<20} {'ВРЕМЯ (с)':<12} {'ЦЕЛЕВАЯ':<15} {'КАМЕРЫ':<12} {'ЭФФ. (%)':<12} {'УЛУЧШ.'}")
        print("-" * 100)

        base_objective = results['Greedy-Balanced']['objective']

        for method, data in results.items():
            if 'quantum_time' in data:
                efficiency = (data['quantum_time'] / data['time'] * 100) if data['time'] > 0 else 0
                improvement = ((base_objective - data['objective']) / base_objective * 100) if base_objective > 0 else 0
                improvement_str = f"{improvement:+.1f}%" if improvement != 0 else "0.0%"

                print(
                    f"{method:<20} {data['time']:<12.2f} {data['objective']:<15.1f} {data['covered']:<12} {efficiency:<12.1f} {improvement_str:<10}")
            else:
                print(
                    f"{method:<20} {data['time']:<12.2f} {data['objective']:<15.1f} {data['covered']:<12} {'-':<12} {'-':<10}")

        print("=" * 100)


def main():
    """Главная функция со сбалансированными настройками"""
    logger.info("=== СБАЛАНСИРОВАННЫЙ ЗАПУСК ===")

    scheduler = BalancedQuantumScheduler(
        n_cameras=5000,
        n_servers=250,
        batch_size=40,
        max_servers_per_batch=15
    )

    utilization = scheduler.generate_realistic_data()
    logger.info(f"Использование емкости: {utilization:.1f}%")

    results = scheduler.run_balanced_comparison()

    logger.info("=== СБАЛАНСИРОВАННЫЙ ЭКСПЕРИМЕНТ ЗАВЕРШЕН ===")
    return results


if __name__ == "__main__":
    results = main()
